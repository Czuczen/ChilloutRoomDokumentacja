164 [HttpPost]
165 public async Task<ActionResult> AbandonQuest(int id)
166 {
167 var quest = await _questRepository.GetAsync(id);
168
169 if (!quest.InProgress) return null;
170
171 var questProgress = await _questRequirementsProgressRepository.SingleAsync(item => item.QuestId == quest.Id);
172 WorkersHelper.ResetQuest(quest, questProgress);
173 await _questService.SetStartValuesAsync(new List<Quest> {quest});
174 var questInfoCreation = await _plantationManager.CreateQuestInfoCreationModel(quest);
175
176 return PartialView("Quests/_quest", ViewModelFactory.CreateQuestInfoViewModel(quest, ObjectMapper, questInfoCreation));
177 }

179 [HttpPost]
180 public async Task<ActionResult> TakeQuest(int id)
181 {
182 var quest = await _questRepository.GetAsync(id);
183
184 if (quest.InProgress) return null;
185
186 var plantationStorage = await _plantationStorageRepository.GetAsync((int) quest.PlantationStorageId);
187 var questInfoCreation = await _plantationManager.CreateQuestInfoCreationModel(quest);
188 await _questService.SetStartValuesAsync(new List<Quest> {quest});
189
190 var limitIsOver = false;
191 var heaveQuestTokens = TokensOperator.QuestTokenFee(_questHub, plantationStorage, quest);
192 if (heaveQuestTokens)
193 limitIsOver = PlantationManagerHelper.CheckQuestLimitsAndSetProgressStatus(quest, plantationStorage, _questHub);
194
195 return !limitIsOver && heaveQuestTokens
196 ? PartialView("Quests/_quest", ViewModelFactory.CreateQuestInfoViewModel(quest, ObjectMapper, questInfoCreation))
197 : null;
198 }
