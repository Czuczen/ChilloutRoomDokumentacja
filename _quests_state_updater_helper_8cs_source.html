26 private static bool NotCurrEntityReq(Requirement req, AnalysisState state)
27 {
28 var ret = true;

30 {
31 if (!string.IsNullOrWhiteSpace(req.CustomEntityName)) // generowane typy połączone - plant, seed, dired fruit

36 else // zwykły generowany typ

44 }
45

243 private static bool AnyImportantDifference(Requirement req, AnalysisState state, out CheckValues checkValues)
244 {
245 var ret = false;
246 checkValues = new CheckValues();
247 var entity = state.Entity;
248 if (state.LastEntityPropertyChanges == null) return false;
249
250 var dbParser = state.DbParser;
251 var propChange = state.LastEntityPropertyChanges.FirstOrDefault(item => item.PropertyName == req.Comparer);
252 if (propChange == null) return false;
253
254 var entityType = entity.GetType();
255 var prop = entityType.GetProperties().ToList()
256 .SingleOrDefault(item => item.Name == propChange.PropertyName);
257 if (prop == null) return false;
258
259 if (req.Condition == DbRequirementConditionsNames.Possession
260 || req.Condition == DbRequirementConditionsNames.Deliver)
261 {
262 var currValue = entityType.GetProperty(req.Comparer)?.GetValue(entity);
263 var parsedValue = dbParser.Parse(prop, currValue);
264 switch (parsedValue)
265 {
266 case int parsedIntValue:
267 checkValues.Int = parsedIntValue;
268 break;
269 case decimal parsedDecimalValue:
270 checkValues.Decimal = parsedDecimalValue;
271 break;
272 }
273
274 ret = true;
275 }
276 else
277 {
278 var newValue = propChange.NewValue;
279 var oldValue = !string.IsNullOrWhiteSpace(propChange.OriginalValue)
280 ? propChange.OriginalValue
281 : "0";
282
283 var parsedOldValue = dbParser.Parse(prop, oldValue);
284 var parsedNewValue = dbParser.Parse(prop, newValue);
285
286 switch (parsedOldValue)
287 {
288 case int parsedIntOldValue when parsedNewValue is int parsedIntNewValue:
289 checkValues.Int = parsedIntNewValue - parsedIntOldValue;
290 break;
291 case decimal parsedDecimalOldValue when parsedNewValue is decimal parsedDecimalNewValue:
292 checkValues.Decimal = parsedDecimalNewValue - parsedDecimalOldValue;
293 break;
294 }
295
296 if ((checkValues.Int != null && checkValues.Int != 0) || (checkValues.Decimal != null && checkValues.Decimal != 0))
297 ret = true;
298 }
299
300 return ret;
301 }

322 private static List<UpdateQuest> GenerateUpdateModel(List<QuestWithRequirements> updatedQuests, List<QuestWithRequirements> allQuestsWithAllRequirements)
323 {
324 var updatingList = new List<UpdateQuest>();
325 foreach (var questWithUpdatedRequirements in updatedQuests)
326 {
327 var updatingQuest = new UpdateQuest
328 {
329 QuestId = questWithUpdatedRequirements.Quest.Id,
330 InProgress = questWithUpdatedRequirements.Quest.InProgress
331 };
332
333 var requirementsProgressDict = allQuestsWithAllRequirements.Single(item => item.Quest.Id == questWithUpdatedRequirements.Quest.Id).RequirementsProgress;
334 foreach (var req in questWithUpdatedRequirements.Requirements)
335 {
336 var updatingReq = new UpdateRequirement
337 {
338 RequirementId = req.Id,
339 RequirementProgressText = requirementsProgressDict[req.Id] + "/" + req.Amount,
340 RequirementProgressPercentage = (requirementsProgressDict[req.Id] / req.Amount * 100 + "%").Replace(",", ".")
341 };
342
343 updatingQuest.UpdatingRequirements.Add(updatingReq);
344 }
345
346 updatingQuest.QuestIsComplete = allQuestsWithAllRequirements
347 .Single(item => item.Quest.Id == questWithUpdatedRequirements.Quest.Id).Requirements
348 .All(item => requirementsProgressDict[item.Id] >= item.Amount);
349
350 updatingList.Add(updatingQuest);
351 }
352
353 return updatingList;
354 }

376 private static void UpdateQuestReq(AnalysisState state, QuestWithRequirements questAnalysis, QuestWithRequirements questWithRequirements, Requirement req,
377 decimal value, EnumUtils.AnalysisTypes analysisType)
378 {
379 var quest = questWithRequirements.Quest;
380 var requirementsProgress = questWithRequirements.RequirementsProgress;
381 var currAmountRequirement = requirementsProgress[req.Id];
382
383 if (req.Condition != DbRequirementConditionsNames.Deliver && req.Condition != DbRequirementConditionsNames.Possession &&
384 currAmountRequirement == req.Amount) return;
385
386 questAnalysis.Quest = quest;
387 questAnalysis.Requirements.Add(req);
388
389 switch (analysisType)
390 {
391 case EnumUtils.AnalysisTypes.ProductOrStorage:
392 switch (req.Condition)
393 {
394 case DbRequirementConditionsNames.Deliver:
395 case DbRequirementConditionsNames.Possession:
396 requirementsProgress[req.Id] = value > req.Amount ? req.Amount : value;
397 break;
398 case DbRequirementConditionsNames.Increase:
399 var increaseCurrAmount = currAmountRequirement + value;
400 requirementsProgress[req.Id] = increaseCurrAmount > req.Amount ? req.Amount : increaseCurrAmount;
401 break;
402 case DbRequirementConditionsNames.Decrease:
403 var decreaseCurrAmount = currAmountRequirement + value * -1;
404 requirementsProgress[req.Id] = decreaseCurrAmount > req.Amount ? req.Amount : decreaseCurrAmount;
405 break;
406 }
407 break;
408 case EnumUtils.AnalysisTypes.Plant:
409 requirementsProgress[req.Id] += value;
410 break;
411 case EnumUtils.AnalysisTypes.BlackMarket:
412 var currAmount = currAmountRequirement + value;
413 requirementsProgress[req.Id] = currAmount > req.Amount ? req.Amount : currAmount;
414 break;
415 default:
416 throw new ArgumentOutOfRangeException(nameof(analysisType), analysisType, null);
417 }
418
419 questWithRequirements.QuestRequirementsProgress.RequirementsProgress = JsonConvert.SerializeObject(requirementsProgress, new DecimalFormatConverter());
420 state.QuestRequirementsProgressRepository.Update(questWithRequirements.QuestRequirementsProgress);
421 }
