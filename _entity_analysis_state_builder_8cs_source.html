257 public virtual AnalysisState Build(EnumUtils.AnalysisBuildTypes type, object entity, string action, EnumUtils.BlackMarketPages? page)
258 {
259 var state = new AnalysisState();
260
261 state.QuestRequirementsProgressRepository = _questRequirementsProgressRepository;
262
263 switch (type)
264 {
265 case EnumUtils.AnalysisBuildTypes.Product:
266 return GetProductAnalysisState(state, (IPlantationGeneratedEntity) entity, action);
267 case EnumUtils.AnalysisBuildTypes.Storage:
268 return GetStorageAnalysisState(state, (PartStorage) entity);
269 case EnumUtils.AnalysisBuildTypes.BlackMarket:
270 return GetBlackMarketAnalysisState(state, (BlackMarketTransaction) entity, page, action);
271 default:
272 throw new ArgumentOutOfRangeException(nameof(type), type, null);
273 }
274 }

284 private AnalysisState GetProductAnalysisState<TEntity>(AnalysisState state, TEntity entity, string action)
285 where TEntity : class, IPlantationGeneratedEntity
286 {
287 if (entity.PlantationStorageId == null) return state;
288
289 state.GeneratedEntity = entity;
290 state.DbParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Db);
291 state.LastEntityPropertyChanges = _entityChangeService.GetLastEntityChangeForEntity(entity)?.PropertyChanges;
292
293 var plantationStorage = _plantationStorageRepository.Get((int) entity.PlantationStorageId);
294 var filteredQuestAndQuestProgress = _questRequirementsProgressRepository.GetAll()
295 .Join(_questRepository.GetAll()
296 .Where(item => item.PlantationStorageId == plantationStorage.Id && !item.IsComplete),
297 progress => progress.QuestId,
298 quest => quest.Id,
299 (progress, quest) => new {progress, quest}).ToList();
300
301 state.PlantationStorage = plantationStorage;
302 state.Quests = filteredQuestAndQuestProgress.Select(item => item.quest).ToList();
303 state.QuestRequirementsProgress = filteredQuestAndQuestProgress.Select(item => item.progress).ToList();
304 state.Requirements = _requirementRepository.GetAllList(item => item.DistrictId == plantationStorage.DistrictId);
305 state.AvailableQuests = EntityAnalysisStateBuilderHelper.GetAvailableQuests(state);
306 state.IgnoreChange = state.Entity is Product product ? _ignoreChangeService.GetIgnoreChangeForUpdatedEntity(product) : null;
307 state.InfoHub = _infoHub;
308 state.BonusHub = _bonusHub;
309 state.QuestHub = _questHub;
310 state.Action = action;
311 state.District = _districtRepository.Get(plantationStorage.DistrictId);
312 state.PlayerStorage = _playerStorageRepository.Single(item => item.UserId == plantationStorage.UserId);
313 state.QuestsWithRequirements = EntityAnalysisStateBuilderHelper.GetQuestsWithRequirements(state);
314
315 if (state.IgnoreChange != null)
316 _ignoreChangeService.Remove(state.IgnoreChange.Id);
317
318 state.InitStatus = true;
319
320 return state;
321 }

330 private AnalysisState GetStorageAnalysisState<TEntity>(AnalysisState state, TEntity entity)
331 where TEntity : PartStorage
332 {
333 state.StorageEntity = entity;
334 state.DbParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Db);
335 state.LastEntityPropertyChanges = _entityChangeService.GetLastEntityChangeForEntity(entity)?.PropertyChanges;
336 state.DisplayParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Display);
337 state.DisplayParserStrategy = new DisplayStrategy();
338 state.ReceivedLevels = EntityAnalysisStateBuilderHelper.GetReceivedLevels(state);
339 state.QuestHub = _questHub;
340 state.InfoHub = _infoHub;
341
342 switch (entity)
343 {
344 case PlantationStorage plantationStorage:
345 var filteredQuestAndQuestProgress = _questRequirementsProgressRepository.GetAll()
346 .Join(_questRepository.GetAll().Where(item => item.PlantationStorageId == plantationStorage.Id && !item.IsComplete),
347 progress => progress.QuestId,
348 quest => quest.Id,
349 (progress, quest) => new {progress, quest}).ToList();
350
351 state.Quests = filteredQuestAndQuestProgress.Select(item => item.quest).ToList();
352 state.QuestRequirementsProgress = filteredQuestAndQuestProgress.Select(item => item.progress).ToList();
353 state.Requirements = _requirementRepository.GetAllList(item => item.DistrictId == plantationStorage.DistrictId);
354 state.District = _districtRepository.Single(item => item.Id == plantationStorage.DistrictId);
355 SetPlantationStorageAnalysisState(state, plantationStorage);
356 break;
357 case PlayerStorage playerStorage:
358 var user = _userRepository.Get(playerStorage.UserId);
359 var filteredQuestAndQuestProgressForPlayerStorage = _questRequirementsProgressRepository.GetAll()
360 .Join(_questRepository.GetAll()
361 .Where(item => item.PlayerStorageId == playerStorage.Id && !item.IsComplete),
362 progress => progress.QuestId,
363 quest => quest.Id,
364 (progress, quest) => new {progress, quest}).ToList();
365
366 state.Quests = filteredQuestAndQuestProgressForPlayerStorage.Select(item => item.quest).ToList();
367 state.QuestRequirementsProgress = filteredQuestAndQuestProgressForPlayerStorage.Select(item => item.progress).ToList();
368 state.Requirements = _requirementRepository.GetAllList();
369 state.PlantationStorages = _plantationStorageRepository.GetAllList(item => item.UserId == playerStorage.UserId);
370 state.AllDistricts = _districtRepository.GetAllList();
371 state.IsDistrictWarden = _permissionChecker.IsGranted(user.ToUserIdentifier(), PermissionNames.Crud_DistrictWarden);
372 EntityAnalysisStateBuilderHelper.SetPlayerStorageAnalysisState(state, playerStorage);
373 break;
374 }
375
376 state.InitStatus = true;
377 return state;

388 private AnalysisState GetBlackMarketAnalysisState(AnalysisState state, BlackMarketTransaction entity, EnumUtils.BlackMarketPages? page, string action)
389 {
390 if (page == EnumUtils.BlackMarketPages.Buyer && entity.BuyerId != null)
391 state.PlantationStorage = _plantationStorageRepository.Single(item => item.UserId == entity.BuyerId && item.DistrictId == entity.DistrictId);
392 else if (page == EnumUtils.BlackMarketPages.Seller && entity.SellerId != null)
393 state.PlantationStorage = _plantationStorageRepository.Single(item => item.UserId == entity.SellerId && item.DistrictId == entity.DistrictId);
394
395 if (state.PlantationStorage == null && page != null)
396 {
397 Logger.Error("Problem wywoany prawdopodobnie w trakcie usuwania poczonych rekordw przez deleter'a.");
398 return state;
399 }
400
401 state.District = _districtRepository.Get(entity.DistrictId);
402 state.DisplayParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Display);
403 state.ObjectMapper = _objectMapper;
404 state.OnlineClientManager = _onlineClientManager;
405 state.BlackMarketHub = _blackMarketHub;
406 state.BlackMarketEntity = entity;
407 state.Action = action;
408
409 if (page != null)
410 {
411 var plantationStorage = state.PlantationStorage;
412 var filteredQuestAndQuestProgress = _questRequirementsProgressRepository.GetAll()
413 .Join(_questRepository.GetAll().Where(item =>
414 item.PlantationStorageId == plantationStorage.Id && !item.IsComplete),
415 progress => progress.QuestId,
416 quest => quest.Id,
417 (progress, quest) => new {progress, quest}).ToList();
418
419 state.QuestHub = _questHub;
420 state.PlayerStorage = _playerStorageRepository.Single(item => item.UserId == plantationStorage.UserId);
421 state.Quests = filteredQuestAndQuestProgress.Select(item => item.quest).ToList();
422 state.QuestRequirementsProgress = filteredQuestAndQuestProgress.Select(item => item.progress).ToList();
423 state.Requirements = _requirementRepository.GetAllList(item => item.DistrictId == plantationStorage.DistrictId);
424 state.AvailableQuests = EntityAnalysisStateBuilderHelper.GetAvailableQuests(state);
425 state.QuestsWithRequirements = EntityAnalysisStateBuilderHelper.GetQuestsWithRequirements(state);
426 }
427
428 state.InitStatus = true;
429
430 return state;
431 }

455 private CurrLvlItems GetReceivedLvlItems(AnalysisState state)
456 {
457 var ret = new CurrLvlItems();
458 var storage = state.PlantationStorage;
459 var oldLvl = state.StorageEntity.Level - state.ReceivedLevels;
460 var lvl = oldLvl;
461
462 for (var i = 0; i < state.ReceivedLevels; i++)
463 {
464 lvl += 1;
465
466 ret.DriedFruits.AddRange(_driedFruitRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
467 ret.Lamps.AddRange(_lampRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
468 ret.Manures.AddRange(_manureRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
469 ret.Pots.AddRange(_potRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
470 ret.Seeds.AddRange(_seedRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
471 ret.Soils.AddRange(_soilRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
472 ret.Waters.AddRange(_waterRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
473 ret.Bonuses.AddRange(_bonusRepository.GetAllList(item => item.PlantationLevelRequirement == lvl && item.PlantationStorageId == storage.Id));
474 ret.Quests.AddRange(state.Quests.Where(item =>
475 item.PlantationLevelRequirement == lvl &&
476 item.IsAvailableInitially &&
477 item.QuestType != DbQuestTypesNames.Achievement).Select(item =>
478 {
479 if (item.QuestType != DbQuestTypesNames.Event) return item;
480 var isCyclic = item.CyclicTime != null && item.CyclicTime > 0;
481 return item.EndTime > Clock.Now || isCyclic ? item : null;
482 }).Where(item => item != null).ToList());
483 }
484
485 _questService.SetStartValues(ret.Quests);
486
487 return ret;
488 }
