84 public override async Task<TenantDto> CreateAsync(CreateTenantDto input)
85 {
86 CheckCreatePermission();
87
88 //Create tenant
89 var tenant = ObjectMapper.Map<Tenant>(input);
90 tenant.ConnectionString = input.ConnectionString.IsNullOrEmpty()
91 ? null
92 : SimpleStringCipher.Instance.Encrypt(input.ConnectionString);
93
94 var defaultEdition = await _editionManager.FindByNameAsync(EditionManager.DefaultEditionName);
95 if (defaultEdition != null)
96 {
97 tenant.EditionId = defaultEdition.Id;
98 }
99
100 await _tenantManager.CreateAsync(tenant);
101 await CurrentUnitOfWork.SaveChangesAsync(); //To get new tenant's id.
102
103 //Create tenant database
104 _abpZeroDbMigrator.CreateOrMigrateForTenant(tenant);
105
106 //We are working entities of new tenant, so changing tenant filter
107 using (CurrentUnitOfWork.SetTenantId(tenant.Id))
108 {
109 //Create static roles for new tenant
110 CheckErrors(await _roleManager.CreateStaticRoles(tenant.Id));
111
112 await CurrentUnitOfWork.SaveChangesAsync(); //To get static role ids
113
114 //grant all permissions to admin role
115 var adminRole = _roleManager.Roles.Single(r => r.Name == StaticRoleNames.Tenants.Admin);
116 await _roleManager.GrantAllPermissionsAsync(adminRole);
117
118 //Create admin user for the tenant
119 var adminUser = User.CreateTenantAdminUser(tenant.Id, input.AdminEmailAddress, User.DefaultPassword);
120 CheckErrors(await _userManager.CreateAsync(adminUser));
121 await CurrentUnitOfWork.SaveChangesAsync(); //To get admin user's id
122
123 //Assign admin user to role!
124 CheckErrors(await _userManager.AddToRoleAsync(adminUser.Id, adminRole.Name));
125 await CurrentUnitOfWork.SaveChangesAsync();
126 }
127
128 return MapToEntityDto(tenant);
129 }
