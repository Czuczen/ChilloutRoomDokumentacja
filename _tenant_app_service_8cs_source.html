47 public override async Task<TenantDto> CreateAsync(CreateTenantDto input)
48 {
49 CheckCreatePermission();
50
51 //Create tenant
52 var tenant = ObjectMapper.Map<Tenant>(input);
53 tenant.ConnectionString = input.ConnectionString.IsNullOrEmpty()
54 ? null
55 : SimpleStringCipher.Instance.Encrypt(input.ConnectionString);
56
57 var defaultEdition = await _editionManager.FindByNameAsync(EditionManager.DefaultEditionName);
58 if (defaultEdition != null)
59 {
60 tenant.EditionId = defaultEdition.Id;
61 }
62
63 await _tenantManager.CreateAsync(tenant);
64 await CurrentUnitOfWork.SaveChangesAsync(); //To get new tenant's id.
65
66 //Create tenant database
67 _abpZeroDbMigrator.CreateOrMigrateForTenant(tenant);
68
69 //We are working entities of new tenant, so changing tenant filter
70 using (CurrentUnitOfWork.SetTenantId(tenant.Id))
71 {
72 //Create static roles for new tenant
73 CheckErrors(await _roleManager.CreateStaticRoles(tenant.Id));
74
75 await CurrentUnitOfWork.SaveChangesAsync(); //To get static role ids
76
77 //grant all permissions to admin role
78 var adminRole = _roleManager.Roles.Single(r => r.Name == StaticRoleNames.Tenants.Admin);
79 await _roleManager.GrantAllPermissionsAsync(adminRole);
80
81 //Create admin user for the tenant
82 var adminUser = User.CreateTenantAdminUser(tenant.Id, input.AdminEmailAddress, User.DefaultPassword);
83 CheckErrors(await _userManager.CreateAsync(adminUser));
84 await CurrentUnitOfWork.SaveChangesAsync(); //To get admin user's id
85
86 //Assign admin user to role!
87 CheckErrors(await _userManager.AddToRoleAsync(adminUser.Id, adminRole.Name));
88 await CurrentUnitOfWork.SaveChangesAsync();
89 }
90
91 return MapToEntityDto(tenant);
92 }
