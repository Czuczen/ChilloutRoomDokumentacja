127 public virtual async Task Buy(int transactionId, bool useDonToken)
128 {
129 try
130 {
131 var status = new BlackMarketTransactionStatus();
132 var userId = AbpSession.GetUserId();
133 var transaction = await _blackMarketTransactionRepository.GetAsync(transactionId);
134 var playerPlantationStorage = await _plantationStorageService.GetPlayerPlantationStorageForLastSelectedDistrictAsync(userId);
135 var district = await _districtRepository.GetAsync(playerPlantationStorage.DistrictId);
136 var cost = transaction.Price * transaction.Quantity;
137
138 var itemRepo = CustomRepositoryFactory.GetRepository(transaction.ItemEntityName);
139 var plantationItems = await itemRepo.GetWhereAsync(RelationFieldsNames.PlantationStorageId, playerPlantationStorage.Id);
140 var playerItem = plantationItems.Cast<Product>().Single(item => item.GeneratedTypeId == transaction.GeneratedTypeId);
141
142 if (useDonToken)
143 {
144 if (!TokensOperator.DonTokenFee(playerPlantationStorage, status))
145 {
146 await Clients.User(userId.ToString()).buy(status);
147 return;
148 }
149 }
150 else if (playerPlantationStorage.Gold < cost)
151 {
152 status = new BlackMarketTransactionStatus {Status = BlackMarketStatuses.DangerStatus, Message = "Za mało kasy"};
153 await Clients.User(userId.ToString()).buy(status);
154 return;
155 }
156 else
157 playerPlantationStorage.Gold -= cost;
158
159 TokensOperator.BlackMarketTokenProfit(district, playerPlantationStorage, cost);
160 if (transaction.SellerId != null) // null jeśli worker wystawia
161 {
162 var sellerPlantationStorage = await _plantationStorageRepository.SingleAsync(item =>
163 item.UserId == transaction.SellerId &&
164 item.DistrictId == playerPlantationStorage.DistrictId);
165 sellerPlantationStorage.Gold += cost;
166 var isDon = await _districtDonRepository.FirstOrDefaultAsync(
167 item => item.DistrictId == sellerPlantationStorage.DistrictId &&
168 item.PlantationStorageId == sellerPlantationStorage.Id) != null;
169
170 TokensOperator.BlackMarketTokenProfit(district, sellerPlantationStorage, cost);
171 TokensOperator.DonTokenProfit(isDon, district, playerPlantationStorage);
172 }
173
174 playerItem.OwnedAmount += transaction.Quantity;
175 status = new BlackMarketTransactionStatus
176 {
177 Status = BlackMarketStatuses.SuccessStatus,
178 Message = "Transakcja zakończona. Kupiono - " + transaction.Quantity +
179 PlantationManagerHelper.GetMeasureUnitByEntityName(transaction.ItemEntityName) +
180 " " + transaction.ItemName
181 };
182
183 transaction.BuyerId = userId;
184 await _blackMarketTransactionRepository.DeleteAsync(transaction.Id);
185
186 await Clients.User(userId.ToString()).buy(status);
187 }
188 catch (Exception ex)
189 {
190 if (ex is EntityNotFoundException exception && exception.EntityType.FullName != null && exception.EntityType.FullName.Contains("BlackMarketTransaction"))
191 {
192 var userId = AbpSession.GetUserId();
193 var data = new BlackMarketTransactionStatus{Status = BlackMarketStatuses.DangerStatus, Message = "Transakcja jest nie dostępna. Upewnij się czy jest jeszcze na liście."};
194 await Clients.User(userId.ToString()).buy(data);
195 }
196 else
197 {
198 Logger.Error("Blad =======//========", ex);
199 await Clients.User(AbpSession.GetUserId().ToString()).errorOccured(ex.Message);
200 }
201 }

210 public virtual async Task Issue(IssueTransaction issueTransaction, bool useBlackMarketToken)
211 {
212 try
213 {
214 var status = new BlackMarketTransactionStatus();
215
216 var userId = AbpSession.GetUserId();
217 var playerName = (await _playerStorageRepository.SingleAsync(item => item.UserId == userId)).PlayerName;
218
219 var repo = CustomRepositoryFactory.GetRepository(issueTransaction.ItemEntityName);
220 var issuedItem = (Product) await repo.GetAsync(issueTransaction.ItemId);
221 var districtId = (await _generatedTypeRepository.GetAsync(issuedItem.GeneratedTypeId)).DistrictId;
222
223 if (issuedItem.OwnedAmount < issueTransaction.Quantity)
224 status = new BlackMarketTransactionStatus{Status = BlackMarketStatuses.DangerStatus, Message = "Niewystarczająca ilość przedmiotu - " + issueTransaction.ItemName};
225 else
226 {
227 var playerPlantationStorage = await _plantationStorageService.GetPlayerPlantationStorageForLastSelectedDistrictAsync(userId);
228 var districtDon = await _districtDonRepository.FirstOrDefaultAsync(item => item.DistrictId == districtId);
229
230 if (districtDon != null)
231 {
232 var donPlantationStorage = await _plantationStorageRepository.SingleAsync(item => item.Id == districtDon.PlantationStorageId);
233 var district = await _districtRepository.GetAsync(districtId);
234 var transactionCost = issueTransaction.Price * issueTransaction.Quantity;
235 var donTribute = district.DonCharityPercentage * transactionCost;
236
237 if (useBlackMarketToken)
238 {
239 if (TokensOperator.DealerTokenFee(playerPlantationStorage, status))
240 {
241 if (TokensOperator.BlackMarketTokenFee(playerPlantationStorage, status))
242 {
243 donPlantationStorage.Gold += donTribute;
244 status = await IssueTransaction(issuedItem, issueTransaction, userId, playerName, districtId);
245 }
246 else
247 {
248 playerPlantationStorage.DealerToken++;
249 await Clients.User(userId.ToString()).issue(status);
250 return;
251 }
252 }
253 else
254 {
255 await Clients.User(userId.ToString()).issue(status);
256 return;
257 }
258 }
259 else if (playerPlantationStorage.Gold < donTribute)
260 {
261 status.Status = BlackMarketStatuses.DangerStatus;
262 status.Message = "Za mało kasy";
263 await Clients.User(userId.ToString()).issue(status);
264 return;
265 }
266 else
267 {
268 if (TokensOperator.DealerTokenFee(playerPlantationStorage, status))
269 {
270 playerPlantationStorage.Gold -= donTribute;
271 donPlantationStorage.Gold += donTribute;
272 status = await IssueTransaction(issuedItem, issueTransaction, userId, playerName, districtId);
273 }
274 else
275 {
276 await Clients.User(userId.ToString()).issue(status);
277 return;
278 }
279 }
280 }
281 else
282 {
283 if (TokensOperator.DealerTokenFee(playerPlantationStorage, status))
284 status = await IssueTransaction(issuedItem, issueTransaction, userId, playerName, districtId);
285 else
286 {
287 await Clients.User(userId.ToString()).issue(status);
288 return;
289 }
290 }
291 }
292
293 await Clients.User(userId.ToString()).issue(status);
294 }
295 catch (Exception ex)
296 {
297 Logger.Error("Blad =======//========", ex);
298 await Clients.User(AbpSession.GetUserId().ToString()).errorOccured(ex.Message);
