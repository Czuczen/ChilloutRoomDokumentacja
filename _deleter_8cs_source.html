186 public virtual async Task DeleteConnected(Requirement entity)
187 {
188 var questsAndQuestsProgress = await _questRequirementsProgressRepository.GetAll().Join(_questRepository.GetAll().Join(
189 _generatedTypeRepository.GetAll().Where(item => item.DistrictId == entity.DistrictId),
190 quest => quest.GeneratedTypeId,
191 type => type.Id,
192 (quest, type) => quest),
193
194 progress => progress.QuestId,
195 quest => quest.Id,
196 (progress, quest) => new {progress, quest}).ToListAsync();
197
198 var quests = questsAndQuestsProgress.Select(item => item.quest).ToList();
199 var questsProgress = questsAndQuestsProgress.Select(item => item.progress).ToList();
200 var questRepo = CustomRepositoryFactory.GetRepository(EntitiesDbNames.Quest);
201
202 foreach (var quest in quests)
203 {
204 var questProgress = questsProgress.Single(item => item.QuestId == quest.Id);
205 var requirementsProgress = JsonConvert.DeserializeObject<Dictionary<int, decimal>>(questProgress.RequirementsProgress);
206 if (!requirementsProgress.ContainsKey(entity.Id)) continue;
207
208 requirementsProgress.Remove(entity.Id);
209
210 if (!requirementsProgress.Keys.Any())
211 {
212 if (quest.PlantationStorageId != null) continue;
213
214 await DeleteDefinitionHelper.DeleteAllDefinitionsAsync(new List<int> {quest.Id}, questRepo,
215 EntitiesDbNames.Quest, _plantService);
216 await _questRepository.DeleteAsync(quest);
217 }
218 else
219 {
220 questProgress.RequirementsProgress = JsonConvert.SerializeObject(requirementsProgress, new DecimalFormatConverter());
221 await _questRequirementsProgressRepository.UpdateAsync(questProgress);
222 }
223 }
