324 public virtual async Task DeleteConnected(Requirement entity)
325 {
326 var questsAndQuestsProgress = await _questRequirementsProgressRepository.GetAll().Join(_questRepository.GetAll().Join(
327 _generatedTypeRepository.GetAll().Where(item => item.DistrictId == entity.DistrictId),
328 quest => quest.GeneratedTypeId,
329 type => type.Id,
330 (quest, type) => quest),
331
332 progress => progress.QuestId,
333 quest => quest.Id,
334 (progress, quest) => new {progress, quest}).ToListAsync();
335
336 var quests = questsAndQuestsProgress.Select(item => item.quest).ToList();
337 var questsProgress = questsAndQuestsProgress.Select(item => item.progress).ToList();
338 var questRepo = CustomRepositoryFactory.GetRepository(EntitiesDbNames.Quest);
339
340 foreach (var quest in quests)
341 {
342 var questProgress = questsProgress.Single(item => item.QuestId == quest.Id);
343 var requirementsProgress = JsonConvert.DeserializeObject<Dictionary<int, decimal>>(questProgress.RequirementsProgress);
344 if (!requirementsProgress.ContainsKey(entity.Id)) continue;
345
346 requirementsProgress.Remove(entity.Id);
347
348 if (!requirementsProgress.Keys.Any())
349 {
350 if (quest.PlantationStorageId != null) continue;
351
352 await DeletePlayerRecordHelper.DeleteAllRecordsByDefs(new List<int> {quest.Id}, questRepo,
353 EntitiesDbNames.Quest, _plantService);
354 await _questRepository.DeleteAsync(quest);
355 }
356 else
357 {
358 questProgress.RequirementsProgress = JsonConvert.SerializeObject(requirementsProgress, new DecimalFormatConverter());
359 await _questRequirementsProgressRepository.UpdateAsync(questProgress);
360 }
361 }
