100 private void SetDuration()
101 {
102 var questsWithDuration = AllQuests.Where(item =>
103 item.PlantationStorageId != null && item.InProgress && item.Duration != null && item.Duration > 0 && item.IsAvailableInitially);
104
105 foreach (var quest in questsWithDuration)
106 {
107 try
108 {
109 var questPlantationStorage = AllPlantationStorages.Single(item => item.Id == (int) quest.PlantationStorageId); // PlantationStorageId nie będzie null. Na tym etapie musi być ustawione.
110 var district = AllDistricts.Single(item => item.Id == questPlantationStorage.DistrictId);
111
112 if (!district.IsDefined) continue;
113 if (quest.PlantationLevelRequirement > questPlantationStorage.Level) continue;
114
115 quest.CurrentDuration++;
116 var playerStorage = AllPlayerStorages.Single(item => item.UserId == questPlantationStorage.UserId);
117 if (questPlantationStorage.DistrictId != playerStorage.LastSelectedDistrictId) continue;
118
119 var userId = questPlantationStorage.UserId;
120 if (quest.Duration <= quest.CurrentDuration)
121 {
122 var questProgress = AllQuestsRequirementsProgress.Single(item => item.QuestId == quest.Id);
123 WorkersHelper.ResetQuest(quest, questProgress);
124 _questService.SetStartValues(new List<Quest> {quest});
125 var obj = new SetDurationResponse
126 {
127 QuestId = quest.Id,
128 RemainingTime = "Pozostały czas: 0sek",
129 TimeUp = true
130 };
131
132 _questHub.Clients.User(userId.ToString()).updateDuration(obj);
133 }
134 else
135 {
136 var remainingTime = quest.Duration - quest.CurrentDuration;
137 var obj = new SetDurationResponse
138 {
139 QuestId = quest.Id,
140 RemainingTime = "Pozostały czas: " + remainingTime + "sek",
141 TimeUp = false
142 };
143
144 _questHub.Clients.User(userId.ToString()).updateDuration(obj);
145 }
146 }
147 catch (Exception ex)
148 {
149 if (ex is InvalidOperationException && ex.Message.Contains("Sequence contains no matching element"))
150 Logger.Warn("Wyjątek wywołany prawdopodobnie w trakcie usuwania połączonych rekordów przez deleter'a. ||||||", ex);
151 else
152 Logger.Error("Blad =======//========", ex);
153 }
154 }

245 var district = AllDistricts.Single(item => item.Id == plantationStorage.DistrictId);
246
247 if (!district.IsDefined) continue;
248 if (eventQuest.PlantationLevelRequirement > plantationStorage.Level) continue;
249
250 var isCyclic = eventQuest.CyclicTime != null && eventQuest.CyclicTime > 0;
251 var playerStorage = AllPlayerStorages.Single(item => item.UserId == plantationStorage.UserId);
252
253 eventQuest.WorkerSent = false;

255

260 eventQuest.IsAvailableInitially = false;
261
262 _questHub.Clients.User(plantationStorage.UserId.ToString()).removeNotAvailableEventQuest(eventQuest.Id, eventQuest.QuestType, eventQuest.IsComplete);
263 }
