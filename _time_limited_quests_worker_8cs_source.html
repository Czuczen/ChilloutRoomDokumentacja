173 private void SetDuration()
174 {
175 var questsWithDuration = AllQuests.Where(item =>
176 item.PlantationStorageId != null && item.InProgress && item.Duration != null && item.Duration > 0 && item.IsAvailableInitially);
177
178 foreach (var quest in questsWithDuration)
179 {
180 try
181 {
182 var questPlantationStorage = AllPlantationStorages.Single(item => item.Id == (int) quest.PlantationStorageId); // PlantationStorageId nie będzie null. Na tym etapie musi być ustawione.
183 var district = AllDistricts.Single(item => item.Id == questPlantationStorage.DistrictId);
184
185 if (!district.IsDefined) continue;
186 if (quest.PlantationLevelRequirement > questPlantationStorage.Level) continue;
187
188 quest.CurrentDuration++;
189 var playerStorage = AllPlayerStorages.Single(item => item.UserId == questPlantationStorage.UserId);
190 if (questPlantationStorage.DistrictId != playerStorage.LastSelectedDistrictId) continue;
191
192 var userId = questPlantationStorage.UserId;
193 if (quest.Duration <= quest.CurrentDuration)
194 {
195 var questProgress = AllQuestsRequirementsProgress.Single(item => item.QuestId == quest.Id);
196 WorkersHelper.ResetQuest(quest, questProgress);
197 _questService.SetStartValues(new List<Quest> {quest});
198 var obj = new SetDurationResponse
199 {
200 QuestId = quest.Id,
201 RemainingTime = "Pozostały czas: 0sek",
202 TimeUp = true
203 };
204
205 _questHub.Clients.User(userId.ToString()).updateDuration(obj);
206 }
207 else
208 {
209 var remainingTime = quest.Duration - quest.CurrentDuration;
210 var obj = new SetDurationResponse
211 {
212 QuestId = quest.Id,
213 RemainingTime = "Pozostały czas: " + remainingTime + "sek",
214 TimeUp = false
215 };
216
217 _questHub.Clients.User(userId.ToString()).updateDuration(obj);
218 }
219 }
220 catch (Exception ex)
221 {
222 if (ex is InvalidOperationException && ex.Message.Contains("Sequence contains no matching element"))
223 Logger.Warn("Wyjątek wywołany prawdopodobnie w trakcie usuwania połączonych rekordów przez deleter'a. ||||||", ex);
224 else
225 Logger.Error("Blad =======//========", ex);
226 }
227 }

317 item.QuestType == DbQuestTypesNames.Event && item.EndTime < currDateTime &&
318 item.PlantationStorageId != null && item.IsAvailableInitially && item.WorkerSent).ToList();

320 foreach (var eventQuest in eventQuestsAfterEvent)

322 try
323 {
324 var plantationStorage = AllPlantationStorages.Single(item => item.Id == (int) eventQuest.PlantationStorageId); // PlantationStorageId nie będzie null. Na tym etapie musi być ustawione.
325 var district = AllDistricts.Single(item => item.Id == plantationStorage.DistrictId);
326
327 if (!district.IsDefined) continue;
328 if (eventQuest.PlantationLevelRequirement > plantationStorage.Level) continue;
329
330 var isCyclic = eventQuest.CyclicTime != null && eventQuest.CyclicTime > 0;
331 var playerStorage = AllPlayerStorages.Single(item => item.UserId == plantationStorage.UserId);
332
333 eventQuest.WorkerSent = false;
334 eventQuest.InProgress = false;
335
336 if ((!isCyclic && eventQuest.IsComplete) || plantationStorage.DistrictId != playerStorage.LastSelectedDistrictId)

340 eventQuest.IsAvailableInitially = false;

342 _questHub.Clients.User(plantationStorage.UserId.ToString()).removeNotAvailableEventQuest(eventQuest.Id, eventQuest.QuestType, eventQuest.IsComplete);
343 }
