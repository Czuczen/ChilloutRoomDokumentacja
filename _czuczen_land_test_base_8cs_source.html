77 /* Uses single database for host and Default tenant,
78 * but dedicated databases for all other tenants.
79 */
80 private void UseDatabasePerTenant()
81 {
82 _hostDb = DbConnectionFactory.CreateTransient();
83 _tenantDbs = new Dictionary<int, DbConnection>();
84
85 LocalIocManager.IocContainer.Register(
86 Component.For<DbConnection>()
87 .UsingFactoryMethod((kernel) =>
88 {
89 lock (_tenantDbs)
90 {
91 var currentUow = kernel.Resolve<ICurrentUnitOfWorkProvider>().Current;
92 var abpSession = kernel.Resolve<IAbpSession>();
93
94 var tenantId = currentUow != null ? currentUow.GetTenantId() : abpSession.TenantId;
95
96 if (tenantId == null || tenantId == 1) //host and default tenant are stored in host db
97 {
98 return _hostDb;
99 }
100
101 if (!_tenantDbs.ContainsKey(tenantId.Value))
102 {
103 _tenantDbs[tenantId.Value] = DbConnectionFactory.CreateTransient();
104 }
105
106 return _tenantDbs[tenantId.Value];
107 }
108 }, true)
109 .LifestyleTransient()
110 );
111 }
