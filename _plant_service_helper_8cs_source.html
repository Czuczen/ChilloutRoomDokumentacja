138 public static int CalculateSetsBuff(List<string> productsSetsNames, List<Bonus> activeBonuses, ExtendingModels.Models.General.District district)
139 {
140 var setsBaf = 1;
141 var bonusSetsBuff = 1;
142
143 var countedSetNames = new List<int>();
144 foreach (var setName in productsSetsNames)
145 {
146 countedSetNames.Add(productsSetsNames.Where(currItem =>
147 !string.IsNullOrWhiteSpace(currItem) &&
148 string.Equals(currItem, setName, StringComparison.CurrentCultureIgnoreCase)).ToList().Count);
149 }
150
151 var oneSixElement = countedSetNames.Any(item => item == 6);
152 var countOneSetElements = countedSetNames.Where(item => item >= 3 && item != 6).ToList().Count;
153 var oneThreeElements = countOneSetElements >= 3 && countOneSetElements != 6;
154 var twoThreeElements = countedSetNames.Where(item => item >= 3 && item != 6).ToList().Count == 6;
155
156 if (oneSixElement)
157 setsBaf = 4;
158
159 if (oneThreeElements)
160 setsBaf = 2;
161
162 if (twoThreeElements)
163 setsBaf = 3;
164
165
166 var bonusesSetsNames = activeBonuses.Select(bonus => bonus.SetName).ToList();
167 var countedBonusesSetsNames = new List<int>();
168 foreach (var setName in bonusesSetsNames)
169 {
170 countedBonusesSetsNames.Add(bonusesSetsNames.Where(currItem =>
171 !string.IsNullOrWhiteSpace(currItem) &&
172 string.Equals(currItem, setName, StringComparison.CurrentCultureIgnoreCase)).ToList().Count);
173 }
174
175 var maxBonusesCount = district.MaxBuffsSlots + district.MaxArtifactSlots;
176 var halfOfMaxBonusesCount = maxBonusesCount / 2;
177
178 var oneMaxElementFromBonus = countedBonusesSetsNames.Any(item => item == maxBonusesCount);
179 var countOneSetElementsFromBonus = countedBonusesSetsNames
180 .Where(item => item >= halfOfMaxBonusesCount && item != maxBonusesCount).ToList().Count;
181 var oneHalfElementsFromBonus = countOneSetElementsFromBonus >= halfOfMaxBonusesCount &&
182 countOneSetElements != maxBonusesCount;
183 var twoHalfElementsFromBonus =
184 countedBonusesSetsNames.Where(item => item >= halfOfMaxBonusesCount && item != maxBonusesCount).ToList()
185 .Count == maxBonusesCount;
186
187 if (oneMaxElementFromBonus)
188 bonusSetsBuff = 4;
189
190 if (oneHalfElementsFromBonus)
191 bonusSetsBuff = 2;
192
193 if (twoHalfElementsFromBonus)
194 bonusSetsBuff = 3;
195
196
197 if (bonusSetsBuff > 1)
198 setsBaf += bonusSetsBuff - 1;
199
200 return setsBaf;
201 }
