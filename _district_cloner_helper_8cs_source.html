48 private static SqlCommandContext GetSqlCommandContext(object entity, string tableName, List<PropertyInfo> entityProperties)
49 {
50 var ret = new SqlCommandContext
51 {
52 Query = $@"
53 SET IDENTITY_INSERT {tableName} ON
54
55 Insert into {tableName}("
56 };
57
58 var valuesFields = @")
59 Values(";
60
61 var sqlParameters = new List<SqlParameter>();
62 foreach (var prop in entityProperties)
63 {
64 var fieldName = "@" + prop.Name;
65 var propIndex = entityProperties.FindIndex(item => item.Name == prop.Name) + 1;
66 var itIsLast = propIndex == entityProperties.Count;
67 var tempObjValue = prop.GetValue(entity);
68 var objValue = tempObjValue ?? DBNull.Value;
69
70 ret.Query += prop.Name + (itIsLast ? "" : ", ");
71 valuesFields += fieldName + (itIsLast ? "" : ", ");
72 sqlParameters.Add(new SqlParameter(fieldName, objValue));
73 }
74
75 ret.Query += $@"{valuesFields})
76
77 SET IDENTITY_INSERT {tableName} OFF";
78
79 ret.SqlParameters = sqlParameters.Cast<object>().ToArray();
80
81 return ret;
82 }

92 public static async Task<List<int>> CloneObjects(AbpDbContext ctx, List<object> objects,
93 List<Tuple<string, int, int>> entityNameOldIdNewIdList, Dictionary<string, int> entitiesMaxIdNumbers)
94 {
95 var ret = new List<int>();
96 if (objects == null || objects.Count <= 0) return ret;
97
98 var objectsType = objects.First().GetType();
99 var entityName = objectsType.Name;
100 var entityProperties = objectsType.GetProperties().ToList();
101 var tableName = GetTableNameByEntityDbName(entityName);
102
103 foreach (var obj in objects)
104 {
105 SetId(obj, entityName, entityNameOldIdNewIdList, entitiesMaxIdNumbers);
106 SetRelationIds(obj, objectsType, entityNameOldIdNewIdList);
107 var command = GetSqlCommandContext(obj, tableName, entityProperties);
108 var status = await ctx.Database.ExecuteSqlCommandAsync(command.Query, command.SqlParameters);
109 if (status == 1)
110 ret.Add(((IEntity<int>) obj).Id);
111 else
112 throw new Exception("Błąd. Status nieprawidłowy. Status: " + status);
113 }
114
115 return ret;

213 Dictionary<string, int> entitiesMaxIdNumbers)
214 {
215 var entity = (IEntity<int>) obj;
216 var oldId = entity.Id;

287 var ret = new Dictionary<string, List<Dictionary<string, object>>>();

290 foreach (var xlsFileFullPath in xlsFilesFromDisk)
291 {
292 var fileName = Path.GetFileNameWithoutExtension(xlsFileFullPath);
293 var entityName = filesIdsAndNames != null ? filesIdsAndNames[fileName] : fileName;
294
295 var stream = File.Open(xlsFileFullPath, FileMode.Open, FileAccess.Read);
296 var excelReader = ExcelReaderFactory.CreateReader(stream);
297 var result = excelReader.AsDataSet(new ExcelDataSetConfiguration{ConfigureDataTable = _ => new ExcelDataTableConfiguration{UseHeaderRow = true}});
298 var dt = result.Tables[0];
299
300 var objectsList = new List<Dictionary<string, object>>();
301 var columnsNames = dt.Columns.Cast<DataColumn>().Select(item => item.ColumnName).ToList();
302 foreach (var row in dt.Rows.Cast<DataRow>())
303 {
304 var obj = new Dictionary<string, object>();
305 foreach (var columnName in columnsNames)
306 obj[columnName] = row[columnName];
307
308 objectsList.Add(obj);
309 }
310
311 ret[entityName] = objectsList;
312 excelReader.Close();
313 }

315 return ret;
