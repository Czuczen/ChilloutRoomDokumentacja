92 private void Buy()
93 {
94 var bought = new List<BlackMarketTransaction>();
95 var groupedByDistrict = AllBlackMarketTransactions.GroupBy(item => item.DistrictId);
96 foreach (var group in groupedByDistrict)
97 {
98 try
99 {
100 var district = AllDistricts.Single(item => item.Id == group.Key);
101 if (!district.IsDefined) continue;
102
103 var districtTransactions = group.ToList();
104 if (districtTransactions.Count <= district.MaxBlackMarketTransactionsCount) continue;
105 foreach (var transaction in districtTransactions)
106 {
107 var randomInt = _random.Next(1, 100);
108 if (randomInt <= district.ChanceForAddBuyBlackMarketTransaction)
109 bought.Add(transaction);
110 }

123 var groupedBought = bought.GroupBy(item => item.DistrictId);
124 foreach (var group in groupedBought)
125 {
126 try
127 {
128 var drawnTransactions = new List<BlackMarketTransaction>();
129 var districtTransactions = group.ToList();
130 var district = AllDistricts.Single(item => item.Id == group.Key);
131
132 for (var i = 0; i < district.NumberDrawnTransactionsToBuy; i++)
133 drawnTransactions.Add(districtTransactions[_random.Next(districtTransactions.Count)]);
134
135 if (district.BuyUsersTransactions)
136 drawnTransactions.AddRange(AllBlackMarketTransactions.Where(item => item.DistrictId == group.Key && item.SellerId != null));
137
138 foreach (var transaction in drawnTransactions.GroupBy(item => item.Id).Select(item => item.First()))
139 {
140 if (transaction.SellerId != null)
141 {
142 var cost = transaction.Price * transaction.Quantity;
143 var plantationStorage = AllPlantationStorages.Single(item =>
144 item.UserId == transaction.SellerId && item.DistrictId == transaction.DistrictId);
145 plantationStorage.Gold += cost;
146 var isDon = AllDistrictDons.FirstOrDefault(
147 item => item.DistrictId == plantationStorage.DistrictId &&
148 item.PlantationStorageId == plantationStorage.Id) != null;
149
150 TokensOperator.BlackMarketTokenProfit(district, plantationStorage, cost);
151 TokensOperator.DonTokenProfit(isDon, district, plantationStorage);
152 }
153
154 _blackMarketTransactionRepository.Delete(transaction.Id);

167 private void Issue()
168 {
169 var toIssued = new List<IBlackMarketWorkerProduct>();
170 foreach (var district in AllDistricts)
171 {
172 if (!district.IsDefined) continue;
173
174 var districtTransactions = AllBlackMarketTransactions.Where(item => item.DistrictId == district.Id).ToList();
175 if (districtTransactions.Count >= district.MaxBlackMarketTransactionsCount + 5) continue;
176 var districtProductsGeneratedTypes = AllGeneratedTypes.Where(item =>
177 item.DistrictId == district.Id && item.EntityName != EntitiesDbNames.Quest);
178
179 foreach (var generatedType in districtProductsGeneratedTypes)
180 {
181 try
182 {
183 var randomIntForBuy = _random.Next(1, 100);
184 if (randomIntForBuy > district.ChanceForAddIssueBlackMarketTransaction) continue;
185
186 var randomInt = _random.Next(1, 100);
187 var seedTypeChance = randomInt <= district.ChanceForIssueSeedAgainstDriedFruitInBlackMarket;
188 var entityName = generatedType.EntityName == EntitiesDbNames.Seed
189 ? seedTypeChance ? generatedType.EntityName : EntitiesDbNames.DriedFruit
190 : generatedType.EntityName;
191
192 var blackMarketItem = (IBlackMarketWorkerProduct) _customRepositoryLoader.GetRepository(entityName)
193 .GetWhere(RelationFieldsNames.GeneratedTypeId, generatedType.Id)
194 .Cast<IPlantationGeneratedEntity>()
195 .SingleOrDefault(item => item.PlantationStorageId == null);
196 if (blackMarketItem == null || !blackMarketItem.IsBlackMarketWorkerItem) continue;
197
198 toIssued.Add(blackMarketItem);
