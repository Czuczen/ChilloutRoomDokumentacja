84 public override async Task DeleteAsync(EntityDto<int> input)

89 if (role.IsStatic)

91 throw new UserFriendlyException("Nie można usunąć roli statycznej");

104 private Task<List<long>> GetUsersInRoleAsync(string roleName)
105 {
106 var users = (from user in _userRepository.GetAll()
107 join userRole in _userRoleRepository.GetAll() on user.Id equals userRole.UserId
108 join role in _roleRepository.GetAll() on userRole.RoleId equals role.Id
109 where role.Name == roleName
110 select user.Id).Distinct().ToList();
111
112 return Task.FromResult(users);
