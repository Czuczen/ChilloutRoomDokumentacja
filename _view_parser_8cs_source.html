64 public async Task<List<object>> ParseObjectsValues(List<object> dtoObjList, List<PropertyInfo> properties)
65 {
66 var ret = new List<Dictionary<string, object>>();
67
68 if (dtoObjList == null || dtoObjList.Count == 0 || properties == null || properties.Count == 0)
69 return ret.Cast<object>().ToList();
70
71 var records = JsonConvert.DeserializeObject<List<Dictionary<string, object>>>(JsonConvert.SerializeObject(dtoObjList));
72 var displayParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Display);
73 var relationFieldsNames = ViewParserHelper.GetRelationFields(properties).Select(item => item.Name).ToList();
74
75 foreach (var obj in records)
76 {
77 var newObj = new Dictionary<string, object>();
78 foreach (var objElem in obj)
79 {
80 var prop = properties.Single(item => item.Name == objElem.Key);
81 if (relationFieldsNames.Any(item => item == objElem.Key))
82 newObj[objElem.Key] = await ParseRelationFieldToName(prop, objElem);
83 else
84 newObj[objElem.Key] = displayParser.Parse(prop, objElem.Value);
85 }
86
87 ret.Add(newObj);
88 }
89
90 return ret.Cast<object>().ToList();
91 }

99 private async Task<string> ParseRelationFieldToName(PropertyInfo prop, KeyValuePair<string, object> objElem)
100 {
101 var ret = "";
102
103 if (objElem.Value == null) return "Brak";
104 if (objElem.Value.ToString() == "") return ret;
105
106 switch (prop.Name)
107 {
108 case RelationFieldsNames.EntityName:
109 return SelectListLoaderHelper.EntitiesNamesDbToHr[objElem.Value.ToString()];
110 case RelationFieldsNames.Condition:
111 return SelectListLoaderHelper.RequirementConditionsNamesDbToHr[objElem.Value.ToString()];
112 case RelationFieldsNames.QuestType:
113 return SelectListLoaderHelper.QuestTypesNamesDbToHr[objElem.Value.ToString()];
114 case RelationFieldsNames.Comparer:
115 return SelectListLoaderHelper.RequirementComparersNamesDbToHr[objElem.Value.ToString()];
116 case RelationFieldsNames.CustomEntityName:
117 return SelectListLoaderHelper.CustomEntityNamesDbToHr[objElem.Value.ToString()];
118 }
119
120 var entityName = ViewParserHelper.GetRelationFieldEntityName(prop);
121 var isUserEntity = entityName == EntitiesDbNames.User;
122 var value = isUserEntity
123 ? (object) long.Parse(objElem.Value.ToString())
124 : (object) int.Parse(objElem.Value.ToString());
125
126 try
127 {
128 ret = isUserEntity
129 ? (await _userRepository.GetAsync((long) value)).Name
130 : await _customRepositoryLoader.GetObjectName(entityName, (int) value);
131 }
132 catch (Exception ex)
133 {
134 if (ex is InvalidOperationException && ex.Message.Contains("Sequence contains no matching element"))
135 {
136 ret = "Nie istnieje";
137 Logger.Warn("Nie istnieje. Encja - " + entityName + ". Id: " + value, ex);
138 }
139 else if (ex is EntityNotFoundException && ex.Message.Contains("There is no such an entity"))
140 {
141 ret = "Nie istnieje";
142 Logger.Warn("Nie istnieje. Encja - " + entityName + ". Id: " + value, ex);
143 }
144 else
145 Logger.Error("Blad =======//========", ex);
146 }
147
148 return ret;
149 }
