56 private async Task<AbpLoginResult<Tenant, User>> GetLoginResultAsync(string usernameOrEmailAddress, string password, string tenancyName)
57 {
58 var loginResult = await _logInManager.LoginAsync(usernameOrEmailAddress, password, tenancyName);

69 private Exception CreateExceptionForFailedLoginAttempt(AbpLoginResultType result, string usernameOrEmailAddress, string tenancyName)
70 {
71 switch (result)
72 {
73 case AbpLoginResultType.Success:
74 return new ApplicationException("Nie wywołuj tej metody z wynikiem sukcesu!");
75 case AbpLoginResultType.InvalidUserNameOrEmailAddress:
76 case AbpLoginResultType.InvalidPassword:
77 return new UserFriendlyException("Logowanie nie powiodło się", "Nieprawidłowa nazwa użytkownika lub hasło");
78 case AbpLoginResultType.InvalidTenancyName:
79 return new UserFriendlyException("Logowanie nie powiodło się", "Nie ma najemcy z nazwą: " + tenancyName);
80 case AbpLoginResultType.TenantIsNotActive:
81 return new UserFriendlyException("Logowanie nie powiodło się", "Najemca nie jest aktywny: " + tenancyName);
82 case AbpLoginResultType.UserIsNotActive:
83 return new UserFriendlyException("Logowanie nie powiodło się", "Użytkownik nie jest aktywny i nie może się zalogować: " + usernameOrEmailAddress);
84 case AbpLoginResultType.UserEmailIsNotConfirmed:
85 return new UserFriendlyException("Logowanie nie powiodło się", "Adres e-mail użytkownika nie jest potwierdzony i nie można się zalogować");
86 default: //Can not fall to default actually. But other result types can be added in the future and we may forget to handle it
87 Logger.Warn("Unhandled login fail reason: " + result);
88 return new UserFriendlyException("Logowanie nie powiodło się");
89 }
