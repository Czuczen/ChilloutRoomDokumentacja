148 public static List<SelectListItem> GetCheckBoxAsSelectOptions(string value)
149 {
150 var ret = new List<SelectListItem>();
151
152 var parsingSuccessful = bool.TryParse(value, out var parsedValue);
153
154 var item = new SelectListItem
155 {
156 Text = null,
157 Value = null,
158 Selected = !parsingSuccessful
159 };
160
161 ret.Add(item);
162
163 var item1 = new SelectListItem
164 {
165 Text = "Tak",
166 Value = "true",
167 Selected = parsedValue && parsingSuccessful
168 };
169
170 ret.Add(item1);
171
172 var item2 = new SelectListItem
173 {
174 Text = "Nie",
175 Value = "false",
176 Selected = !parsedValue && parsingSuccessful
177 };
178
179 ret.Add(item2);
180
181 return ret;
182 }
