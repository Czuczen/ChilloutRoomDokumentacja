172 public async Task DeletePlantAndSetUseCountConnectedProducts(string fieldToCompare, int value, bool needIgnoreChange = true)
173 {
174 var plantType = typeof(ExtendingModels.Models.General.Plant);
175 var allPlants = await _plantRepository.GetAllListAsync();
176
177 foreach (var plant in allPlants)
178 {
179 try
180 {
181 var fieldToCompareValue = (int?) plantType.GetProperty(fieldToCompare)?.GetValue(plant);
182 if (fieldToCompareValue != value) continue;
183
184 await _plantRepository.DeleteAsync(plant.Id);
185
186 var plantPot = await _potRepository.GetAsync(plant.PotId);
187 plantPot.InUseCount--;
188 plantPot.OwnedAmount++;
189
190 var plantLamp = await _lampRepository.GetAsync(plant.LampId);
191 plantLamp.InUseCount--;
192 plantLamp.OwnedAmount++;
193
194 var plantSoil = await _soilRepository.GetAsync(plant.SoilId);
195 var soilPenaltyCount = PlantServiceHelper.CountSoilPenalty(plantPot);
196 var soilReturnCount = plantPot.Capacity - soilPenaltyCount;
197 plantSoil.OwnedAmount += soilReturnCount;
198
199 if (!needIgnoreChange) continue;
200
201 await _ignoreChangeService.Add(plantPot);
202 await _ignoreChangeService.Add(plantLamp);
203 await _ignoreChangeService.Add(plantSoil);
204 }
205 catch (Exception ex)
206 {
207 Logger.Error("Blad======//=======", ex);
208 }
209 }
210 }
