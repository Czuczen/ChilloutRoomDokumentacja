58 public async Task<Dictionary<string, object>> GetSelectFieldsOptions(List<PropertyInfo> properties, List<object> values, string entity, bool isEdit, bool isAdmin)
59 {
60 var ret = new Dictionary<string, object>();
61 var relationFields = ViewParserHelper.GetRelationFields(properties);
62
63 foreach (var prop in relationFields)
64 {
65 ret[prop.Name] = new List<SelectListItem>();
66 try
67 {
68 switch (prop.Name)
69 {
70 case RelationFieldsNames.GeneratedTypeId:
71 {
72 var fieldIndex = properties.FindIndex(item => item.Name == RelationFieldsNames.GeneratedTypeId);
73 var fieldValue = values?.Count > 0 ? (int?) values[fieldIndex] : null;
74 var availableGeneratedTypes = await GetAvailableGeneratedTypes(entity, fieldValue);
75 ret[prop.Name] = SelectListLoaderHelper.GetGeneratedTypesAsSelectOptions(availableGeneratedTypes, fieldValue);
76 break;
77 }
78 case RelationFieldsNames.EntityName:
79 ret[prop.Name] = SelectListLoaderHelper.GetGeneratedEntitiesAsSelectOptions(properties, values);
80 break;
81 case RelationFieldsNames.QuestType:
82 ret[prop.Name] = SelectListLoaderHelper.GetQuestTypesAsSelectOptions(properties, values);
83 break;
84 case RelationFieldsNames.Condition:
85 ret[prop.Name] = SelectListLoaderHelper.GetRequirementConditionsAsSelectOptions(properties, values);
86 break;
87 case RelationFieldsNames.Comparer:
88 ret[prop.Name] = SelectListLoaderHelper.GetRequirementComparersAsSelectOptions(properties, values);
89 break;
90 case RelationFieldsNames.CustomEntityName:
91 ret[prop.Name] = SelectListLoaderHelper.GetCustomEntitiesAsSelectOptions(properties, values);
92 break;
93 case RelationFieldsNames.UserId:
94 {
95 var users = await GetAvailableUsersForSelectOptions(entity, isEdit, properties, values, isAdmin);
96 ret[prop.Name] = SelectListLoaderHelper.GetUsersAsSelectOptions(users, properties, values);
97 break;
98 }
99 case RelationFieldsNames.DistrictId when !isAdmin:
100 {
101 var userId = AbpSession.GetUserId();
102 var district = (await _districtRepository.GetAllListAsync(currDistrict => currDistrict.UserId == userId)).SingleOrDefault();
103 if (district == null) break;
104
105 ret[prop.Name] = new List<SelectListItem>
106 {
107 new() {Text = district.Name, Value = district.Id.ToString(), Selected = true}
108 };
109 break;
110 }
111 default:
112 {
113 var entityName = ViewParserHelper.GetRelationFieldEntityName(prop);
114 var allItems = (await _customRepositoryLoader.GetRepository(entityName).GetAllListAsync()).Cast<INamedEntity>();
115 var fieldIndex = properties.FindIndex(item => item.Name == prop.Name);
116 var options = allItems.Select(currObj => new SelectListItem
117 {
118 Text = currObj.Name,
119 Value = currObj.Id.ToString(),
120 Selected = values?.Count > 0 && currObj.Id == (int?) values[fieldIndex]
121 }).ToList();
122
123 ret[prop.Name] = options;
124 break;
125 }
126 }
127 }
128 catch (Exception ex)
129 {
130 Logger.Error("Blad=====/=====", ex);
131 }
132 }
133
134 return ret;
135 }

137 private async Task<List<GeneratedType>> GetAvailableGeneratedTypes(string entity, int? updatingObjGeneratedTypeid)
138 {
139 var ret = new List<GeneratedType>();
140 var entityEnum = SelectListLoaderHelper.GetEntityEnum(entity);
141 var filteredGeneratedTypes = await GetFilteredGeneratedTypes(entity);
142
143 if (entity == EntitiesDbNames.Drop || entity == EntitiesDbNames.Requirement) return filteredGeneratedTypes;
144
145 foreach (var generatedType in filteredGeneratedTypes)
146 {
147 var typeEntityEnum = SelectListLoaderHelper.GetEntityEnum(generatedType.EntityName);
148 if (entityEnum == EnumUtils.Entities.DriedFruit && typeEntityEnum == EnumUtils.Entities.Seed)
149 {
150 var driedFruitUseThisType = (await _driedFruitRepository.GetAllListAsync(item =>
151 item.GeneratedTypeId == generatedType.Id && item.PlantationStorageId == null)).SingleOrDefault();
152
153 if (driedFruitUseThisType == null)
154 ret.Add(generatedType);
155 }
156 else
157 {
158 var typeIsUsed = false;
159 var allItems = (await _customRepositoryLoader.GetRepository(generatedType.EntityName).GetAllListAsync()).Cast<IGeneratedEntity>().ToList();
160 foreach (var item in allItems)
161 if (generatedType.Id == item.GeneratedTypeId && generatedType.Id != updatingObjGeneratedTypeid)
162 typeIsUsed = true;
163
164 if (!typeIsUsed)
165 ret.Add(generatedType);
166 }
167 }
168
169 return ret;
170 }

172 private async Task<List<User>> GetAvailableUsersForSelectOptions(string entity, bool isEdit, List<PropertyInfo> properties, List<object> values, bool currUserIsAdmin)
173 {
174 var ret = new List<User>();
175
176 if (isEdit)
177 {
178 // Aby uytkownik, ktry jest ju na edytowanym obiekcie te by sugerowany
179 var userIdPropIndex = properties.FindIndex(item => item.Name == RelationFieldsNames.UserId);
180 var userId = (long) values[userIdPropIndex];
181 try
182 {
183 var userFromObject = await _userRepository.GetAsync(userId);
184 ret.Add(userFromObject);
185 }
186 catch (Exception ex)
187 {
188 if (ex is EntityNotFoundException && ex.Message.Contains("There is no such an entity"))
189 Logger.Warn("Nie istnieje. Encja - " + entity + ". Id: " + userId, ex);
190 else
191 Logger.Error("Blad =======//========", ex);
192 }
193 }
194
195 if (currUserIsAdmin && entity == EntitiesDbNames.District)
196 {
197 var allUsers = await _userRepository.GetAllListAsync();
198 var allDistricts = await _districtRepository.GetAllListAsync();
199
200 // Wszyscy opiekunowie, ktrzy nie s przypisani do dzielnicy
201 foreach (var user in allUsers)
202 {
203 var isAdmin = await _permissionChecker.IsGrantedAsync(user.ToUserIdentifier(), PermissionNames.Crud_Admin);
204 var isDistrictWarden = await _permissionChecker.IsGrantedAsync(user.ToUserIdentifier(), PermissionNames.Crud_DistrictWarden);
205 var userDistrict = allDistricts.SingleOrDefault(item => item.UserId == user.Id);
206 if (userDistrict == null && isDistrictWarden && !isAdmin)
207 ret.Add(user);
208 }
209 }
210
211 return ret;
212 }

214 private async Task<List<GeneratedType>> GetFilteredGeneratedTypes(string entity)
215 {
216 var ret = new List<GeneratedType>();
217
218 var userId = AbpSession.GetUserId();
219 var wardenDistrict = (await _districtRepository.GetAllListAsync(item => item.UserId == userId)).SingleOrDefault();
220 if (wardenDistrict == null) return ret;
221
222 if (entity == EntitiesDbNames.Drop || entity == EntitiesDbNames.Requirement)
223 {
224 ret = await _generatedTypeRepository.GetAllListAsync(item => item.DistrictId == wardenDistrict.Id);
225 }
226 else
227 {
228 ret = await _generatedTypeRepository.GetAllListAsync(item =>
229 (item.EntityName == entity && item.DistrictId == wardenDistrict.Id)
230 ||
231 item.EntityName == EntitiesDbNames.Seed && entity == EntitiesDbNames.DriedFruit
232 && item.DistrictId == wardenDistrict.Id);
233 }
234
235 return ret;
