40 public static MvcHtmlString CreateProductInfo(object product)
41 {
42 var ret = "";
43
44 if (product == null)
45 return MvcHtmlString.Create(ret);
46
47 var generatedType = product.GetType();
48 var nameProp = generatedType.GetProperty(EntitiesFieldsDbNames.Name);
49 var productProperties = generatedType.GetProperties();
50 var propsValues = generatedType.GetProperties().Select(currProp => currProp.GetValue(product)).ToList();
51 var propsNames = PlantationManagerHelper.GetHrPropList(generatedType);
52 var displayParser = ParserFactory.GetParser(EnumUtils.ParseStrategies.Display);
53 var propName = propsNames.SingleOrDefault(item => item == ProductFieldsHrNames.Name);
54
55 if (propName == ProductFieldsHrNames.Name)
56 {
57 var nameIndex = propsNames.FindIndex(item => item == ProductFieldsHrNames.Name);
58 var validNameFieldValue = displayParser.Parse(nameProp, propsValues[nameIndex]);
59 ret += $"<span>{propName}: {validNameFieldValue}</span>";
60 }
61
62 for (var i = 0; i < propsNames.Count; i++)
63 {
64 if (NotRenderedProductInfoFields.Any(currName => currName == propsNames[i])) continue;
65
66 var validFieldValue = displayParser.Parse(productProperties[i], propsValues[i]);
67 ret += $"<span>{propsNames[i]}: {validFieldValue}</span>";
68 }
69
70 return MvcHtmlString.Create(ret);
71 }
