152 public virtual async Task<ShopTransaction> ResourcesTransaction(int productId, string entity, decimal howMany, string action)
153 {
154 try
155 {
156 var ret = new ShopTransaction();
157 var userId = AbpSession.GetUserId();
158 var playerStorage = await _playerStorageService.GetPlayerStorage(userId);
159 var playerPlantationStorage = await _plantationStorageService.GetPlayerPlantationStorageForLastSelectedDistrictAsync(userId);
160 var district = await _districtRepository.GetAsync(playerPlantationStorage.DistrictId);
161
162 if (PlantationStorageObservedFields.CurrencyExchanges.Any(item => item == entity))
163 StorageHubHelper.ProcessCurrencyExchange(district, playerPlantationStorage, playerStorage, entity, action, howMany, ret);
164 else
165 await ProcessProduct(playerPlantationStorage, action, howMany, ret, productId, entity);
166
167 return ret;
168 }
169 catch (Exception ex)
170 {
171 Logger.Error("Blad =======//========", ex);
172 await Clients.User(AbpSession.GetUserId().ToString()).errorOccured(ex.Message);
173 return null;
174 }
175 }
