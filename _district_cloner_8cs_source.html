225 public async Task<DistrictContext> Clone(int districtId)
226 {
227 var ret = await GetDistrictContext(districtId);
228
229 var localRoot = Path.GetFullPath(Path.Combine(AppDomain.CurrentDomain.BaseDirectory));
230 var contentPath = Path.Combine(localRoot, "Content");
231 var definitionsPath = Path.Combine(contentPath, "Definitions");
232 var exportPath = Path.Combine(definitionsPath, "Export");
233 if (!Directory.Exists(exportPath))
234 Directory.CreateDirectory(exportPath);
235
236 var objects = new Dictionary<string, List<object>>
237 {
238 [EntitiesDbNames.District] = new() {ret.District},
239 [EntitiesDbNames.GeneratedType] = ret.GeneratedTypes.Cast<object>().ToList(),
240 [EntitiesDbNames.DriedFruit] = ret.DriedFruits.Cast<object>().ToList(),
241 [EntitiesDbNames.Lamp] = ret.Lamps.Cast<object>().ToList(),
242 [EntitiesDbNames.Manure] = ret.Manures.Cast<object>().ToList(),
243 [EntitiesDbNames.Pot] = ret.Pots.Cast<object>().ToList(),
244 [EntitiesDbNames.Seed] = ret.Seeds.Cast<object>().ToList(),
245 [EntitiesDbNames.Soil] = ret.Soils.Cast<object>().ToList(),
246 [EntitiesDbNames.Water] = ret.Waters.Cast<object>().ToList(),
247 [EntitiesDbNames.Drop] = ret.Drops.Cast<object>().ToList(),
248 [EntitiesDbNames.Quest] = ret.Quests.Cast<object>().ToList(),
249 [EntitiesDbNames.Bonus] = ret.Bonuses.Cast<object>().ToList(),
250 [EntitiesDbNames.Requirement] = ret.Requirements.Cast<object>().ToList(),
251 [EntitiesDbNames.QuestRequirementsProgress] = ret.QuestsRequirementsProgress.Cast<object>().ToList(),
252 [EntitiesDbNames.DropQuest] = ret.DropsQuests.Cast<object>().ToList()
253 };
254
255 foreach (var pair in objects)
256 {
257 if (pair.Value == null || !pair.Value.Any()) continue;
258 var lines = new List<string>();
259
260 var properties = pair.Value.First().GetType().GetProperties();
261 var columnNames = properties.Select(item => item.Name);
262
263 var header = string.Join(",", columnNames.Select(name => $"\"{name}\""));
264 var valueLines = pair.Value
265 .Select(obj => string.Join(",",
266 properties.Select(prop =>
267 {
268 var val = pair.Key == EntitiesDbNames.QuestRequirementsProgress // W QuestRequirementsProgresses jedno pole zawiera json
269 ? prop.GetValue(obj)?.ToString().Replace("\"", "'")
270 : prop.GetValue(obj);
271 return $"\"{val}\"";
272 }))).ToList();
273
274 lines.Add(header);
275 lines.AddRange(valueLines);
276
277 var fullPath = Path.Combine(exportPath, pair.Key);
278 File.WriteAllLines(fullPath + ".xls", lines, Encoding.UTF8);
279 }
280
281 return ret;
282 }
