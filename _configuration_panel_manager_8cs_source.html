56 public Logs FormatLogs()
57 {
58 var ret = new Logs();
59
60 var localRoot = Path.GetFullPath(Path.Combine(AppDomain.CurrentDomain.BaseDirectory));
61 var appDataPath = Path.Combine(localRoot, "App_Data");
62 var logsPath = Path.Combine(appDataPath, "Logs");
63 var filePath = Path.Combine(logsPath, "Logs.txt");
64
65 using (var fs = new FileStream(filePath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite))
66 using (var sr = new StreamReader(fs, Encoding.Default))
67 {
68 var buffer = new char[(int) sr.BaseStream.Length];
69 sr.Read(buffer, 0, (int) sr.BaseStream.Length);
70
71 var rawLogs = new string(buffer);
72 var stringSeparators = new[] { "\r\n" };
73 var lines = rawLogs.Split(stringSeparators, StringSplitOptions.None);
74
75 var log = new List<string>();
76 foreach (var line in lines)
77 {
78 var isNewLog = false;
79 if (line.StartsWith("INFO"))
80 {
81 isNewLog = true;
82
83 log = new List<string> {line};
84 ret.InfoLogs.Add(log);
85 }
86
87 if (line.StartsWith("DEBUG"))
88 {
89 isNewLog = true;
90
91 log = new List<string> {line};
92 ret.DebugLogs.Add(log);
93 }
94
95 if (line.StartsWith("WARN"))
96 {
97 isNewLog = true;
98
99 log = new List<string> {line};
100 ret.WarnLogs.Add(log);
101 }
102
103 if (line.StartsWith("ERROR"))
104 {
105 isNewLog = true;
106
107 log = new List<string> {line};
108 ret.ErrorLogs.Add(log);
109 }
110
111 if (!isNewLog)
112 log.Add(line);
113 }
114 }
115
116 ret.InfoLogs.Reverse();
117 ret.DebugLogs.Reverse();
118 ret.WarnLogs.Reverse();
119 ret.ErrorLogs.Reverse();
120
121 return ret;
122 }

126 public async Task<EditOrCreate> EditOrCreateObject(string entity, IObjectMapper objectMapper, bool isAdmin, int? objectId, string editManyAction)
127 {
128 var ret = new EditOrCreate();
129
130 var isEdit = objectId != null;
131 var obj = isEdit ? await CustomRepositoryFactory.GetRepository(entity).GetAsync(objectId) : null;
132 var entityType = isEdit || !string.IsNullOrWhiteSpace(editManyAction)
133 ? UpdateDtoModelFactory.GetUpdateDtoEntityTypeByEntityDbName(entity)
134 : CreateDtoModelFactory.GetCreateDtoEntityTypeByEntityCreateDtoName(entity);
135 var properties = entityType.GetProperties().ToList();
136
137 ret.EditManyAction = editManyAction;
138 ret.IsEdit = isEdit;
139 ret.EntityName = entity;
140 ret.EntityHrName = SelectListLoaderHelper.EntitiesNamesDbToHr.SingleOrDefault(currItem => currItem.Key == entity).Value;
141 ret.Properties = properties;
142
143 if (isEdit)
144 {
145 ret.ObjectId = objectId;
146 var mappedObj = objectMapper.Map(obj, UpdateDtoModelFactory.GetUpdateDtoNewObjectByEntityUpdateDtoName(entityType.Name));
147 foreach (var prop in properties)
148 ret.EditingObjValues.Add(prop.GetValue(mappedObj));
149 }
150
151 ret.FieldsOptions = await _selectListLoader.GetSelectFieldsOptions(
152 ret.Properties, ret.EditingObjValues, ret.EntityName, ret.IsEdit, isAdmin);
153 ret.EditingObjValues = isEdit
154 ? ParserFactory.GetParser(EnumUtils.ParseStrategies.Edit).Parse(ret.EditingObjValues, properties)
155 : new List<object>();
156
157 await LoadSelectors(ret, objectId, obj);
158
159 return ret;
160 }

162 private async Task LoadSelectors(EditOrCreate editOrCreate, int? objectId, object obj)
163 {
164 editOrCreate.ConnectionSelectors = ConfigurationPanelManagerHelper.GetEntityConnections(editOrCreate.EntityName);
165 if (editOrCreate.IsEdit)
166 {
167 var drops = new List<Drop>();
168 var requirements = new List<Requirement>();
169 foreach (var connection in editOrCreate.ConnectionSelectors)
170 {
171 switch (connection.Key)
172 {
173 case EntitiesDbNames.Drop:
174 var dropOptions = new List<object>();
175 var relations = await _dropQuestRepository.GetAllListAsync(item => item.QuestId == objectId);
176 var dropsIds = relations.Select(item => item.DropId);
177 foreach (var dropId in dropsIds)
178 {
179 var drop = await _dropRepository.GetAsync(dropId);
180 drops.Add(drop);
181 var option = new Select2Data{ Id = drop.Id, Text = drop.Name };
182 dropOptions.Add(option);
183 }
184
185 editOrCreate.ExistingConnections[connection.Key] = dropOptions;
186 break;
187 case EntitiesDbNames.Requirement:
188 var requirementOptions = new List<object>();
189 var questProgress =
190 await _questRequirementsProgressRepository.SingleAsync(item =>
191 item.QuestId == ((Quest) obj).Id);
192 var requirementsProgress = JsonConvert.DeserializeObject<Dictionary<int, decimal>>(questProgress.RequirementsProgress);
193
194 foreach (var reqProgress in requirementsProgress)
195 {
196 var req = await _requirementRepository.GetAsync(reqProgress.Key);
197 requirements.Add(req);
198 var option = new Select2Data{ Id = req.Id, Text = req.Name };
199 requirementOptions.Add(option);
200 }
201
202 editOrCreate.ExistingConnections[connection.Key] = requirementOptions;
203 break;
204 }
205 }
206
207 editOrCreate.ExistingConnectionsDistrictId = drops.FirstOrDefault()?.DistrictId ?? requirements.FirstOrDefault()?.DistrictId;
208 }
209 }

211 public async Task<List<object>> SearchSuggestions(string valueToSearch, string entity, int? generatedTypeId, int? districtId)
212 {
213 var ret = new List<object>();
214
215 int currDistrictId;
216 if (districtId != null)
217 currDistrictId = (int) districtId;
218 else if (generatedTypeId != null)
219 currDistrictId = (await _generatedTypeRepository.GetAsync((int) generatedTypeId)).DistrictId;
220 else
221 throw new AbpException("Brak DistrictId i GeneratedTypeId. Jedno z nich musi istnie");
222
223 switch (entity)
224 {
225 case EntitiesDbNames.Drop:
226 ret = (await _dropRepository.GetAllListAsync(item => item.DistrictId == currDistrictId)).Where(
227 item => string.IsNullOrWhiteSpace(valueToSearch) ||
228 item.Name.ToLower().Contains(valueToSearch.ToLower())).Cast<object>().ToList();
229 break;
230 case EntitiesDbNames.Requirement:
231 ret = (await _requirementRepository.GetAllListAsync(item => item.DistrictId == currDistrictId))
232 .Where(item => string.IsNullOrWhiteSpace(valueToSearch) ||
233 item.Name.ToLower().Contains(valueToSearch.ToLower())).Cast<object>().ToList();
234 break;
235 }
236
237 return ret;
238 }
