100 public Logs FormatLogs()
101 {
102 var ret = new Logs();
103
104 var localRoot = Path.GetFullPath(Path.Combine(AppDomain.CurrentDomain.BaseDirectory));
105 var appDataPath = Path.Combine(localRoot, "App_Data");
106 var logsPath = Path.Combine(appDataPath, "Logs");
107 var filePath = Path.Combine(logsPath, "Logs.txt");
108
109 using (var fs = new FileStream(filePath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite))
110 using (var sr = new StreamReader(fs, Encoding.Default))
111 {
112 var buffer = new char[(int) sr.BaseStream.Length];
113 sr.Read(buffer, 0, (int) sr.BaseStream.Length);
114
115 var rawLogs = new string(buffer);
116 var stringSeparators = new[] { "\r\n" };
117 var lines = rawLogs.Split(stringSeparators, StringSplitOptions.None);
118
119 var log = new List<string>();
120 foreach (var line in lines)
121 {
122 var isNewLog = false;
123 if (line.StartsWith("INFO"))
124 {
125 isNewLog = true;
126
127 log = new List<string> {line};
128 ret.InfoLogs.Add(log);
129 }
130
131 if (line.StartsWith("DEBUG"))
132 {
133 isNewLog = true;
134
135 log = new List<string> {line};
136 ret.DebugLogs.Add(log);
137 }
138
139 if (line.StartsWith("WARN"))
140 {
141 isNewLog = true;
142
143 log = new List<string> {line};
144 ret.WarnLogs.Add(log);
145 }
146
147 if (line.StartsWith("ERROR"))
148 {
149 isNewLog = true;
150
151 log = new List<string> {line};
152 ret.ErrorLogs.Add(log);
153 }
154
155 if (!isNewLog)
156 log.Add(line);
157 }
158 }
159
160 ret.InfoLogs.Reverse();
161 ret.DebugLogs.Reverse();
162 ret.WarnLogs.Reverse();
163 ret.ErrorLogs.Reverse();
164
165 return ret;
166 }

185 public async Task<EditOrCreate> EditOrCreateObject(string entity, IObjectMapper objectMapper, bool isAdmin, int? objectId, string editManyAction)
186 {
187 var ret = new EditOrCreate();
188
189 var isEdit = objectId != null;
190 var obj = isEdit ? await CustomRepositoryFactory.GetRepository(entity).GetAsync(objectId) : null;
191 var entityType = isEdit || !string.IsNullOrWhiteSpace(editManyAction)
192 ? UpdateDtoModelFactory.GetUpdateDtoEntityTypeByEntityDbName(entity)
193 : CreateDtoModelFactory.GetCreateDtoEntityTypeByEntityCreateDtoName(entity);
194 var properties = entityType.GetProperties().ToList();
195
196 ret.EditManyAction = editManyAction;
197 ret.IsEdit = isEdit;
198 ret.EntityName = entity;
199 ret.EntityHrName = SelectListLoaderHelper.EntitiesNamesDbToHr.SingleOrDefault(currItem => currItem.Key == entity).Value;
200 ret.Properties = properties;
201
202 if (isEdit)
203 {
204 ret.ObjectId = objectId;
205 var mappedObj = objectMapper.Map(obj, UpdateDtoModelFactory.GetUpdateDtoNewObjectByEntityUpdateDtoName(entityType.Name));
206 foreach (var prop in properties)
207 ret.EditingObjValues.Add(prop.GetValue(mappedObj));
208 }
209
210 ret.FieldsOptions = await _selectListLoader.GetSelectFieldsOptions(
211 ret.Properties, ret.EditingObjValues, ret.EntityName, ret.IsEdit, isAdmin);
212 ret.EditingObjValues = isEdit
213 ? ParserFactory.GetParser(EnumUtils.ParseStrategies.Edit).Parse(ret.EditingObjValues, properties)
214 : new List<object>();
215
216 await LoadSelectors(ret, objectId, obj);
217
218 return ret;
219 }

227 private async Task LoadSelectors(EditOrCreate editOrCreate, int? objectId, object obj)
228 {
229 editOrCreate.ConnectionSelectors = ConfigurationPanelManagerHelper.GetEntityConnections(editOrCreate.EntityName);
230
231 if (editOrCreate.IsEdit)
232 {
233 var drops = new List<Drop>();
234 var requirements = new List<Requirement>();
235 foreach (var connection in editOrCreate.ConnectionSelectors)
236 {
237 switch (connection.Key)
238 {
239 case EntitiesDbNames.Drop:
240 var dropOptions = new List<object>();
241 var relations = await _dropQuestRepository.GetAllListAsync(item => item.QuestId == objectId);
242 var dropsIds = relations.Select(item => item.DropId);
243 foreach (var dropId in dropsIds)
244 {
245 var drop = await _dropRepository.GetAsync(dropId);
246 drops.Add(drop);
247 var option = new Select2Data{ Id = drop.Id, Text = drop.Name };
248 dropOptions.Add(option);
249 }
250
251 editOrCreate.ExistingConnections[connection.Key] = dropOptions;
252 break;
253 case EntitiesDbNames.Requirement:
254 var requirementOptions = new List<object>();
255 var questProgress =
256 await _questRequirementsProgressRepository.SingleAsync(item =>
257 item.QuestId == ((Quest) obj).Id);
258 var requirementsProgress = JsonConvert.DeserializeObject<Dictionary<int, decimal>>(questProgress.RequirementsProgress);
259
260 foreach (var reqProgress in requirementsProgress)
261 {
262 var req = await _requirementRepository.GetAsync(reqProgress.Key);
263 requirements.Add(req);
264 var option = new Select2Data{ Id = req.Id, Text = req.Name };
265 requirementOptions.Add(option);
266 }
267
268 editOrCreate.ExistingConnections[connection.Key] = requirementOptions;
269 break;
270 }
271 }
272
273 editOrCreate.ExistingConnectionsDistrictId = drops.FirstOrDefault()?.DistrictId ?? requirements.FirstOrDefault()?.DistrictId;
274 }
275 }

285 public async Task<List<object>> SearchSuggestions(string valueToSearch, string entity, int? generatedTypeId, int? districtId)
286 {
287 var ret = new List<object>();
288
289 int currDistrictId;
290 if (districtId != null)
291 currDistrictId = (int) districtId;
292 else if (generatedTypeId != null)
293 currDistrictId = (await _generatedTypeRepository.GetAsync((int) generatedTypeId)).DistrictId;
294 else
295 throw new AbpException("Brak DistrictId i GeneratedTypeId. Jedno z nich musi istnieć");
296
297 switch (entity)
298 {
299 case EntitiesDbNames.Drop:
300 ret = (await _dropRepository.GetAllListAsync(item => item.DistrictId == currDistrictId)).Where(
301 item => string.IsNullOrWhiteSpace(valueToSearch) ||
302 item.Name.ToLower().Contains(valueToSearch.ToLower())).Cast<object>().ToList();
303 break;
304 case EntitiesDbNames.Requirement:
305 ret = (await _requirementRepository.GetAllListAsync(item => item.DistrictId == currDistrictId))
306 .Where(item => string.IsNullOrWhiteSpace(valueToSearch) ||
307 item.Name.ToLower().Contains(valueToSearch.ToLower())).Cast<object>().ToList();
308 break;
309 }
310
311 return ret;
312 }
