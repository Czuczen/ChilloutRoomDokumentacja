84 private void SendDistrictEndTime()
85 {
86 var currDateTime = Clock.Now;
87 var districts = DefinedDistricts.Where(item => item.EndTime != null && item.EndTime < currDateTime).ToList();
88 var districtsWithFutureStartTime = DefinedDistricts.Where(item => item.StartTime > currDateTime).ToList();
89 foreach (var district in districtsWithFutureStartTime)

91 var districtPlantationStorages = PlantationStorages.Where(item =>
92 item.DistrictId == district.Id && item.UserId != district.UserId).ToList();

95 }
96
97 foreach (var district in districts)
98 {
99 try
100 {
101 var districtPlantationStorages = PlantationStorages.Where(item =>
102 item.DistrictId == district.Id && item.UserId != district.UserId).ToList();
103 var districtPlantationStoragesWithPrestige = districtPlantationStorages.Where(item => item.Prestige > 0).ToList();
104
105 if (!districtPlantationStorages.Any()) continue;
106
107 if (districtPlantationStoragesWithPrestige.Any())
108 {
109 var topThree = districtPlantationStoragesWithPrestige.OrderByDescending(item => item.Prestige).Take(3).ToList();
110 if (topThree.Count == 1)
111 {
112 var firstPlace = topThree.Single();
113 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
114 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
115 }
116 else if (topThree.Count == 2)
117 {
118 var firstPlace = topThree.First();
119 var secondPlace = topThree.Last();
120
121 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
122 var secondPlayerStorage = PlayerStorages.Single(item => item.UserId == secondPlace.UserId);
123
124 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
125 secondPlayerStorage.Honor += district.HonorForTakingSecondPlace;
126 }
127 else
128 {
129 var firstPlace = topThree.Aggregate((i1, i2) => i1.Prestige > i2.Prestige ? i1 : i2);
130 var thirdPlace = topThree.Aggregate((i1, i2) => i1.Prestige < i2.Prestige ? i1 : i2);
131 var secondPlace = topThree.Single(item => item.Id != firstPlace.Id && item.Id != thirdPlace.Id);
132
133 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
134 var secondPlayerStorage = PlayerStorages.Single(item => item.UserId == secondPlace.UserId);
135 var thirdPlayerStorage = PlayerStorages.Single(item => item.UserId == thirdPlace.UserId);
136
137 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
138 secondPlayerStorage.Honor += district.HonorForTakingSecondPlace;
139 thirdPlayerStorage.Honor += district.HonorForTakingThirdPlace;
140 }
141
142 foreach (var plantationStorage in districtPlantationStoragesWithPrestige)
143 {
144 var playerStorage = PlayerStorages.Single(item => item.UserId == plantationStorage.UserId);
145 playerStorage.Honor += district.HonorConsolationPrize;
