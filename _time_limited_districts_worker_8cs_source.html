134 private void SendDistrictEndTime()
135 {
136 var currDateTime = Clock.Now;
137 var districts = DefinedDistricts.Where(item => item.EndTime != null && item.EndTime < currDateTime).ToList();
138 var districtsWithFutureStartTime = DefinedDistricts.Where(item => item.StartTime > currDateTime).ToList();
139 foreach (var district in districtsWithFutureStartTime)

141 var districtPlantationStorages = PlantationStorages.Where(item =>
142 item.DistrictId == district.Id && item.UserId != district.UserId).ToList();
143 foreach (var plantationStorage in districtPlantationStorages)

147 foreach (var district in districts)
148 {
149 try
150 {
151 var districtPlantationStorages = PlantationStorages.Where(item =>
152 item.DistrictId == district.Id && item.UserId != district.UserId).ToList();
153 var districtPlantationStoragesWithPrestige = districtPlantationStorages.Where(item => item.Prestige > 0).ToList();
154
155 if (!districtPlantationStorages.Any()) continue;
156
157 if (districtPlantationStoragesWithPrestige.Any())

159 var topThree = districtPlantationStoragesWithPrestige.OrderByDescending(item => item.Prestige).Take(3).ToList();
160 if (topThree.Count == 1)
161 {
162 var firstPlace = topThree.Single();
163 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
164 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
165 }
166 else if (topThree.Count == 2)
167 {
168 var firstPlace = topThree.First();
169 var secondPlace = topThree.Last();
170
171 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
172 var secondPlayerStorage = PlayerStorages.Single(item => item.UserId == secondPlace.UserId);
173
174 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
175 secondPlayerStorage.Honor += district.HonorForTakingSecondPlace;
176 }
177 else
178 {
179 var firstPlace = topThree.Aggregate((i1, i2) => i1.Prestige > i2.Prestige ? i1 : i2);
180 var thirdPlace = topThree.Aggregate((i1, i2) => i1.Prestige < i2.Prestige ? i1 : i2);
181 var secondPlace = topThree.Single(item => item.Id != firstPlace.Id && item.Id != thirdPlace.Id);
182
183 var firstPlayerStorage = PlayerStorages.Single(item => item.UserId == firstPlace.UserId);
184 var secondPlayerStorage = PlayerStorages.Single(item => item.UserId == secondPlace.UserId);
185 var thirdPlayerStorage = PlayerStorages.Single(item => item.UserId == thirdPlace.UserId);

187 firstPlayerStorage.Honor += district.HonorForTakingFirstPlace;
188 secondPlayerStorage.Honor += district.HonorForTakingSecondPlace;
