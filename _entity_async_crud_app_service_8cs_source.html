177 var isUpdateVisible = request.Ids != null && request.Ids.Count > 0;
178 var isUpdateAll = request.Ids == null || request.Ids.Count == 0;
179
180 if (isUpdateVisible)
181 {
182 foreach (var id in request.Ids)

184 var itemToUpdate = ObjectMapper.Map<TUpdateInput>(await GetEntityByIdAsync(id));
185 var updatedItem = (TUpdateInput) AsyncCrudHelper.UpdateObject<TUpdateInput>(request.FieldsToUpdate, itemToUpdate);
186 ResponseBuilder.AddItems(await UpdateAsync(updatedItem));
187 }
188 }
189 else if (isUpdateAll)

191 var allItems = ObjectMapper.Map<List<TUpdateInput>>(await Repository.GetAllListAsync());
192 foreach (var itemToUpdate in allItems)

194 var updatedItem = (TUpdateInput) AsyncCrudHelper.UpdateObject<TUpdateInput>(request.FieldsToUpdate, itemToUpdate);
195 ResponseBuilder.AddItems(await UpdateAsync(updatedItem));
