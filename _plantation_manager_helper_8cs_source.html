243 public static bool CheckQuestLimitsAndSetProgressStatus(Quest quest, PlantationStorage plantationStorage, IHubContext questHub)
244 {
245 var ret = false;
246 var currDateTime = Clock.Now;
247 var userId = plantationStorage.UserId;
248
249 switch (quest.QuestType)
250 {
251 case DbQuestTypesNames.Achievement:
252 case DbQuestTypesNames.Others:
253 quest.InProgress = true;
254 break;
255 case DbQuestTypesNames.Daily:
256 if (plantationStorage.StartedDailyQuestsCount < plantationStorage.UnlockedDailyQuestsCount)
257 {
258 quest.InProgress = true;
259 plantationStorage.StartedDailyQuestsCount++;
260 }
261 else
262 {
263 questHub.Clients.User(userId.ToString()).theLimitExpired(quest.Id, quest.QuestType, "Dzienny limit wykorzystany");
264 ret = true;
265 }
266 break;
267 case DbQuestTypesNames.Weekly:
268 if (plantationStorage.StartedWeeklyQuestsCount < plantationStorage.UnlockedWeeklyQuestsCount)
269 {
270 quest.InProgress = true;
271 plantationStorage.StartedWeeklyQuestsCount++;
272 }
273 else
274 {
275 questHub.Clients.User(userId.ToString()).theLimitExpired(quest.Id, quest.QuestType, "Tygodniowy limit wykorzystany");
276 ret = true;
277 }
278 break;
279 case DbQuestTypesNames.Event:
280 if (quest.StartTime <= currDateTime && quest.EndTime > currDateTime)
281 {
282 quest.InProgress = true;
283 }
284 else
285 {
286 var phrase = quest.EndTime < Clock.Now ? "byo" : "bdzie";
287 var message = "Zadanie " + phrase +" dostpne od " + quest.StartTime + " do " + quest.EndTime;
288 questHub.Clients.User(userId.ToString()).theLimitExpired(quest.Id, quest.QuestType, message);
289 ret = true;
290 }
291 break;
292 }
293
294 return ret;

412 public static void AddLevels(District district, PlantationStorage playerPlantationStorage, decimal gainedExp, List<string> dropsNotification, PlayerStorage playerStorage,
413 bool isManyLevel = false, decimal tempPlantGainedExp = decimal.MinValue, decimal tempExpToNexLevel = decimal.MinValue)
414 {

416 {
417 playerPlantationStorage.Level++;
418 if (district.EndTime == null)
419 playerStorage.Level++;
420
421 tempPlantGainedExp = isManyLevel ? tempPlantGainedExp - tempExpToNexLevel : (playerPlantationStorage.CurrExp + gainedExp) - playerPlantationStorage.ExpToNextLevel;
422 tempExpToNexLevel = isManyLevel ? tempExpToNexLevel * 2 : playerPlantationStorage.ExpToNextLevel * 2;
423

425 {

427 }
428 else
429 {
430 playerPlantationStorage.CurrExp = tempPlantGainedExp;
431 playerPlantationStorage.ExpToNextLevel = tempExpToNexLevel;
432
433 playerStorage.GainedExperience += gainedExp;
434 playerPlantationStorage.GainedExperience += gainedExp;
435
436 dropsNotification.Add("Otrzymano " + gainedExp + "pkt dowiadczenia");
437 }
438 }
439 else
440 {
441 playerStorage.GainedExperience += gainedExp;
