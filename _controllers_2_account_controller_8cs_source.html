129 private async Task SignInAsync(User user, ClaimsIdentity identity = null, bool rememberMe = false)
130 {
131 if (identity == null)
132 {
133 identity = await _userManager.CreateIdentityAsync(user, DefaultAuthenticationTypes.ApplicationCookie);
134 }
135
136 _authenticationManager.SignOut(DefaultAuthenticationTypes.ApplicationCookie);
137 // Many browsers do not clean up session cookies when you close them. So the rule of thumb must be:
138 // For having a consistent behaviour across all browsers, don't rely solely on browser behaviour for proper clean-up
139 // of session cookies. It is safer to use non-session cookies (IsPersistent == true) in bundle with an expiration date.
140 // See http://blog.petersondave.com/cookies/Session-Cookies-in-Chrome-Firefox-and-Sitecore/
141 if (rememberMe) {
142 _authenticationManager.SignIn(new AuthenticationProperties { IsPersistent = true }, identity);
143 } else {
144 _authenticationManager.SignIn(
145 new AuthenticationProperties
146 {
147 IsPersistent = true,
148 ExpiresUtc = DateTimeOffset.UtcNow.AddMinutes(int.Parse(System.Configuration.ConfigurationManager.AppSettings["AuthSession.ExpireTimeInMinutes.WhenNotPersistent"] ?? "1440"))
149 },
150 identity);
151 }

154 private Exception CreateExceptionForFailedLoginAttempt(AbpLoginResultType result, string usernameOrEmailAddress, string tenancyName)
155 {
156 switch (result)
157 {
158 case AbpLoginResultType.Success:
159 return new ApplicationException("Nie wywołuj tej metody z wynikiem sukcesu!");
160 case AbpLoginResultType.InvalidUserNameOrEmailAddress:
161 case AbpLoginResultType.InvalidPassword:
162 return new UserFriendlyException("Logowanie nie powiodło się", "Nieprawidłowa nazwa użytkownika lub hasło");
163 case AbpLoginResultType.InvalidTenancyName:
164 return new UserFriendlyException("Logowanie nie powiodło się", "Nie ma najemcy z nazwą: " + tenancyName);
165 case AbpLoginResultType.TenantIsNotActive:
166 return new UserFriendlyException("Logowanie nie powiodło się", "Najemca nie jest aktywny: " + tenancyName);
167 case AbpLoginResultType.UserIsNotActive:
168 return new UserFriendlyException("Logowanie nie powiodło się", "Użytkownik nie jest aktywny i nie może się zalogować: " + usernameOrEmailAddress);
169 case AbpLoginResultType.UserEmailIsNotConfirmed:
170 return new UserFriendlyException("Logowanie nie powiodło się", "Adres e-mail użytkownika nie jest potwierdzony i nie można się zalogować");
171 case AbpLoginResultType.LockedOut:
172 return new UserFriendlyException("Logowanie nie powiodło się", "Wiadomość zablokowana przez użytkownika");
173 default: //Can not fall to default actually. But other result types can be added in the future and we may forget to handle it
174 Logger.Warn("Unhandled login fail reason: " + result);
175 return new UserFriendlyException("Logowanie nie powiodło się");
176 }

211 [HttpPost]
212 public virtual async Task<ActionResult> Register(RegisterViewModel model)
213 {
214 try
215 {
216 CheckModelState();
217
218 //Create user
219 var user = new User
220 {
221 Name = model.Name,
222 Surname = model.Surname,
223 EmailAddress = model.EmailAddress,
224 IsActive = true
225 };
226
227 //Get external login info if possible
228 ExternalLoginInfo externalLoginInfo = null;
229 if (model.IsExternalLogin)
230 {
231 externalLoginInfo = await _authenticationManager.GetExternalLoginInfoAsync();
232 if (externalLoginInfo == null)
233 {
234 throw new ApplicationException("Nie można zalogować się z zewnątrz!");
235 }
236
237 user.Logins = new List<UserLogin>
238 {
239 new UserLogin
240 {
241 LoginProvider = externalLoginInfo.Login.LoginProvider,
242 ProviderKey = externalLoginInfo.Login.ProviderKey
243 }
244 };
245
246 if (model.UserName.IsNullOrEmpty())
247 {
248 model.UserName = model.EmailAddress;
249 }
250
251 model.Password = Authorization.Users.User.CreateRandomPassword();
252
253 if (string.Equals(externalLoginInfo.Email, model.EmailAddress, StringComparison.InvariantCultureIgnoreCase))
254 {
255 user.IsEmailConfirmed = true;
256 }
257 }
258 else
259 {
260 //Username and Password are required if not external login
261 if (model.UserName.IsNullOrEmpty() || model.Password.IsNullOrEmpty())
262 {
263 throw new UserFriendlyException("Formularz jest nieprawidłowy");
264 }
265 }
266
267 user.UserName = model.UserName;
268 user.Password = new PasswordHasher().HashPassword(model.Password);
269
270 //Switch to the tenant
271 _unitOfWorkManager.Current.EnableFilter(AbpDataFilters.MayHaveTenant); //TODO: Needed?
272 _unitOfWorkManager.Current.SetTenantId(AbpSession.GetTenantId());
273
274 //Add default roles
275 user.Roles = new List<UserRole>();
276 foreach (var defaultRole in await _roleManager.Roles.Where(r => r.IsDefault).ToListAsync())
277 {
278 user.Roles.Add(new UserRole { RoleId = defaultRole.Id });
279 }
280
281 //Save user
282 CheckErrors(await _userManager.CreateAsync(user));
283 await _unitOfWorkManager.Current.SaveChangesAsync();
284
285 //Directly login if possible
286 if (user.IsActive)
287 {
288 AbpLoginResult<Tenant, User> loginResult;
289 if (externalLoginInfo != null)
290 {
291 loginResult = await _logInManager.LoginAsync(externalLoginInfo.Login, GetTenancyNameOrNull());
292 }
293 else
294 {
295 loginResult = await GetLoginResultAsync(user.UserName, model.Password, GetTenancyNameOrNull());
296 }
297
298 if (loginResult.Result == AbpLoginResultType.Success)
299 {
300 await SignInAsync(loginResult.User, loginResult.Identity);
301 return Redirect(Url.Action("Index", "Home"));
302 }
303
304 Logger.Warn("New registered user could not be login. This should not be normally. login result: " + loginResult.Result);
305 }
306
307 //If can not login, show a register result page
308 return View("RegisterResult", new RegisterResultViewModel
309 {
310 TenancyName = GetTenancyNameOrNull(),
311 NameAndSurname = user.Name + " " + user.Surname,
312 UserName = user.UserName,
313 EmailAddress = user.EmailAddress,
314 IsActive = user.IsActive
315 });
316 }
317 catch (UserFriendlyException ex)
318 {
319 ViewBag.IsMultiTenancyEnabled = _multiTenancyConfig.IsEnabled;
320 ViewBag.ErrorMessage = ViewHelper.GetCorrectMessage(ex, model);
321
322 return View("Register", model);
323 }
324 }

349 public virtual async Task<ActionResult> ExternalLoginCallback(string returnUrl, string tenancyName = "")
350 {
351 var loginInfo = await _authenticationManager.GetExternalLoginInfoAsync();
352 if (loginInfo == null)
353 {
354 return RedirectToAction("Login");
355 }
356
357 //Try to find tenancy name
358 if (tenancyName.IsNullOrEmpty())
359 {
360 var tenants = await FindPossibleTenantsOfUserAsync(loginInfo.Login);
361 switch (tenants.Count)
362 {
363 case 0:
364 return await RegisterView(loginInfo);
365 case 1:
366 tenancyName = tenants[0].TenancyName;
367 break;
368 default:
369 return View("TenantSelection", new TenantSelectionViewModel
370 {
371 Action = Url.Action("ExternalLoginCallback", "Account", new { returnUrl }),
372 Tenants = ObjectMapper.Map<List<TenantSelectionViewModel.TenantInfo>>(tenants)
373 });
374 }
375 }
376
377 var loginResult = await _logInManager.LoginAsync(loginInfo.Login, tenancyName);
378
379 switch (loginResult.Result)
380 {
381 case AbpLoginResultType.Success:
382 await SignInAsync(loginResult.User, loginResult.Identity, false);
383
384 if (string.IsNullOrWhiteSpace(returnUrl))
385 {
386 returnUrl = Url.Action("Index", "Home");
387 }
388
389 return Redirect(returnUrl);
390 case AbpLoginResultType.UnknownExternalLogin:
391 return await RegisterView(loginInfo, tenancyName);
392 default:
393 throw CreateExceptionForFailedLoginAttempt(loginResult.Result, loginInfo.Email ?? loginInfo.DefaultUserName, tenancyName);
394 }
395 }

434 private static bool TryExtractNameAndSurnameFromClaims(List<Claim> claims, ref string name, ref string surname)
435 {
436 string foundName = null;
437 string foundSurname = null;
438
439 var givennameClaim = claims.FirstOrDefault(c => c.Type == ClaimTypes.GivenName);
440 if (givennameClaim != null && !givennameClaim.Value.IsNullOrEmpty())
441 {
442 foundName = givennameClaim.Value;
443 }
444
445 var surnameClaim = claims.FirstOrDefault(c => c.Type == ClaimTypes.Surname);
446 if (surnameClaim != null && !surnameClaim.Value.IsNullOrEmpty())
447 {
448 foundSurname = surnameClaim.Value;
449 }
450
451 if (foundName == null || foundSurname == null)
452 {
453 var nameClaim = claims.FirstOrDefault(c => c.Type == ClaimTypes.Name);
454 if (nameClaim != null)
455 {
456 var nameSurName = nameClaim.Value;
457 if (!nameSurName.IsNullOrEmpty())
458 {
459 var lastSpaceIndex = nameSurName.LastIndexOf(' ');
460 if (lastSpaceIndex < 1 || lastSpaceIndex > (nameSurName.Length - 2))
461 {
462 foundName = foundSurname = nameSurName;
463 }
464 else
465 {
466 foundName = nameSurName.Substring(0, lastSpaceIndex);
467 foundSurname = nameSurName.Substring(lastSpaceIndex);
468 }
469 }
470 }
471 }
472
473 if (!foundName.IsNullOrEmpty())
474 {
475 name = foundName;
476 }
477
478 if (!foundSurname.IsNullOrEmpty())
479 {
480 surname = foundSurname;
481 }
482
483 return foundName != null && foundSurname != null;
484 }
