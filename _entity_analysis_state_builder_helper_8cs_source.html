152 public static int GetReceivedLevels(AnalysisState state)
153 {
154 var lastPropertiesChanges = state.LastEntityPropertyChanges;
155 var properties = state.Entity.GetType().GetProperties();
156
157 if (lastPropertiesChanges == null || !lastPropertiesChanges.Any()) return 0;
158
159 var lvlPropertyChange = lastPropertiesChanges.SingleOrDefault(item => item.PropertyName == PlantationStorageObservedFields.Level);
160
161 if (lvlPropertyChange?.OriginalValue == null) return 0;
162
163 var prop = properties.SingleOrDefault(item => item.Name == lvlPropertyChange.PropertyName);
164 var dbParser = state.DbParser;
165 var parsedOldValue = (int) dbParser.Parse(prop, lvlPropertyChange.OriginalValue);
166 var parsedNewValue = (int) dbParser.Parse(prop, lvlPropertyChange.NewValue);
167
168 return parsedNewValue - parsedOldValue;
169 }
