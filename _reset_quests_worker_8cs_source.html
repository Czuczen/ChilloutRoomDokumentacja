128 var weeklyLimitTimeHasPassed = weeklySubtractedDate >= DateOfFirstDayWeek;

130 var questQuery = _questRepository.GetAll().Where(item =>

136 progress => progress.QuestId,
137 quest => quest.Id,
138 (progress, quest) => new {progress, quest}).ToList();
139
140 var allDistricts = _districtRepository.GetAllList();
141 var allPlantationStorages = _plantationStorageRepository.GetAllList();
142
143 var completedQuests = filteredQuestAndQuestProgress.Select(item => item.quest).ToList();
144 var questsProgress = filteredQuestAndQuestProgress.Select(item => item.progress).ToList();
145
146 _questHub ??= GlobalHost.ConnectionManager.GetHubContext<QuestHub>();
147
148 if (completedQuests.Any())
149 {
150 foreach (var quest in completedQuests)
151 {
152 try
153 {
154 var playerPlantationStorage = allPlantationStorages.Single(item => item.Id == quest.PlantationStorageId);
155 var district = allDistricts.Single(item => item.Id == playerPlantationStorage.DistrictId);
156 if (!district.IsDefined) continue;
157
158 var questProgress = questsProgress.Single(item => item.QuestId == quest.Id);
159 switch (quest.QuestType)
160 {
161 case DbQuestTypesNames.Daily:
162 WorkersHelper.ResetQuest(quest, questProgress);
163 _questService.SetStartValues(new List<Quest> {quest});
164 _questHub.Clients.User(playerPlantationStorage.UserId.ToString()).setRestartedQuest(quest.Id, quest.QuestType);
165 break;
166 case DbQuestTypesNames.Weekly:
167 if (weeklyLimitTimeHasPassed)
168 {
169 WorkersHelper.ResetQuest(quest, questProgress);
170 _questService.SetStartValues(new List<Quest> {quest});
171 _questHub.Clients.User(playerPlantationStorage.UserId.ToString()).setRestartedQuest(quest.Id, quest.QuestType);
172 }
173 break;
174 }
175 }

179 Logger.Warn("Wyjątek wywołany prawdopodobnie w trakcie usuwania połączonych rekordów przez deleter'a. ||||||", ex);
