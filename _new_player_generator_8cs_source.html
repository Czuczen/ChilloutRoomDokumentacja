165 public async Task<Plantation> GetOrInitPlayerResources(bool heWantPayForHollow, PlayerStorage playerStorage, User user)
166 {
167 var ret = new Plantation();
168
169 var dateTimeNow = Clock.Now;
170 var aWeekBeforeStart = dateTimeNow + TimeSpan.FromDays(7); // informacje o dzielnicy będą widoczne tydzień przed jej startem
171
172 var lastSelectedDistrictId = playerStorage.LastSelectedDistrictId;
173 var allDistricts = await _districtRepository.GetAllListAsync();
174 ret.PlayerStorage = playerStorage;
175
176 ret.AvailableDistricts = allDistricts.Where(item =>
177 (item.StartTime <= aWeekBeforeStart && item.IsDefined &&
178 item.EndTime == null) ||
179 (item.StartTime <= aWeekBeforeStart && item.IsDefined &&
180 item.EndTime != null && item.EndTime > dateTimeNow)).ToList(); // jeśli opiekun dzielnicy chce testować dzielnicę musi ustawić ją na isDefined = true żeby pracowały workery ale jeśli nie chce aby użytkownicy widzieli info o dzielnicy StartTime musi ustawi co najmniej na tydzień do przodu
181
182 var adminPermission = await _permissionChecker.IsGrantedAsync(user.ToUserIdentifier(), PermissionNames.Crud_Admin);
183 var wardenPermission = await _permissionChecker.IsGrantedAsync(user.ToUserIdentifier(), PermissionNames.Crud_DistrictWarden);
184 var wardenDistrict = allDistricts.SingleOrDefault(item => item.UserId == playerStorage.UserId);
185
186 var isDistrictWarden = wardenPermission && wardenDistrict != null;
187 var onlyWardenPermission = wardenPermission && wardenDistrict == null;
188
189 var districts = adminPermission || onlyWardenPermission
190 ? new List<District>()
191 : isDistrictWarden
192 ? new List<District> {wardenDistrict}
193 : allDistricts.Where(item =>
194 (item.StartTime <= dateTimeNow && item.IsDefined &&
195 item.PlayerLevelRequirement <= playerStorage.Level &&
196 item.EndTime == null) ||
197 (item.StartTime <= dateTimeNow && item.IsDefined &&
198 item.PlayerLevelRequirement <= playerStorage.Level &&
199 item.EndTime != null && item.EndTime > dateTimeNow)).ToList();
200
201 if (lastSelectedDistrictId == null || lastSelectedDistrictId == 0)
202 {
203 ret.IsNewPlayer = true;
204 ret.S2DistrictsList = NewPlayerGeneratorHelper.GetS2DistrictsList(districts, new List<District>());
205 }
206 else
207 await GetOrInitForSelectedDistrict(ret, heWantPayForHollow, isDistrictWarden, districts, allDistricts);
208
209 return ret;
210 }

220 private async Task GetOrInitForSelectedDistrict(Plantation plantation, bool heWantPayForHollow, bool isDistrictWarden, List<District> districts, List<District> allDistricts)
221 {
222 plantation.SelectedDistrictId = plantation.PlayerStorage.LastSelectedDistrictId;
223
224 try
225 {
226 plantation.District = await _districtRepository.GetAsync((int) plantation.SelectedDistrictId);
227 }
228 catch (Exception ex)
229 {
230 if (ex is EntityNotFoundException && ex.Message.Contains("There is no such an entity"))
231 {
232 plantation.DistrictLoadingMessage = "Ostatnio wybrana dzielnica została usunięta";
233 Logger.Warn("Ostatnio wybrana dzielnica została usunięta.", ex);
234 }
235 else
236 Logger.Error("Blad =======//========", ex);
237 }
238
239 var district = plantation.District;
240 if (district != null)
241 {
242 if (district.PlayerLevelRequirement <= plantation.PlayerStorage.Level)
243 {
244 await CreateAndSetHollowIfPossible(heWantPayForHollow, isDistrictWarden, plantation);
245 await SetS2DistrictsList(districts, plantation, allDistricts);
246 }
247 else
248 {
249 plantation.TooLowLevel = true;
250 plantation.DistrictLoadingMessage = "Zbyt niski poziom gracza. Wymagany poziom to " + district.PlayerLevelRequirement;
251 }
252 }
253 else
254 {
255 await SetS2DistrictsList(districts, plantation, allDistricts);
256 }

265 private async Task CreateAndSetHollowIfPossible(bool heWantPayForHollow, bool isDistrictWarden, Plantation plantation)
266 {
267 var clockNow = Clock.Now;
268 var district = plantation.District;
269
270 if (!district.IsDefined && !isDistrictWarden)
271 {
272 plantation.DistrictLoadingMessage = "Dzielnica wyłączona";
273 return;
274 }
275
276 if (district.EndTime < clockNow && !isDistrictWarden)
277 {
278 plantation.DistrictLoadingMessage = "Czas dzielnicy minął";
279 return;
280 }
281
282 if (district.StartTime > clockNow && !isDistrictWarden)
283 {
284 plantation.DistrictLoadingMessage = "Czas dzielnicy jeszcze nie nadszedł";
285 return;
286 }
287
288 plantation.PlantationStorage = (await _plantationStorageRepository.GetAllListAsync(item =>
289 item.DistrictId == district.Id && item.UserId == plantation.PlayerStorage.UserId)).SingleOrDefault();
290
291 if (plantation.PlantationStorage == null && (district.HollowPrice == 0 || heWantPayForHollow))
292 {
293 var hollowPrice = district.HollowPrice;
294
295 if (plantation.PlayerStorage.Gold >= hollowPrice)
296 await CreateHollow(plantation);
297 else
298 {
299 plantation.DistrictLoadingMessage = "Nie stać Cię";
300 plantation.HollowPrice = hollowPrice;
301 }
302 }
303 else if (plantation.PlantationStorage == null)
304 {
305 var hollowCost = new DisplayStrategy().ParseValue(EnumUtils.PropTypes.Decimal, district.HollowPrice);
306
307 plantation.MustBuyHollow = true;
308 plantation.DistrictLoadingMessage = "Chcesz kupić dziuple w dzielnicy " + district.Name + " za " + hollowCost + "$?";
309 plantation.HollowPrice = district.HollowPrice;
310 }
311 }

421 private static async Task<List<object>> GenerateRecordsForNewPlayerPlantationStorage<TEntity>(object definitions,
422 EnumUtils.Entities? entityEnum, Plantation plantation, IRepository<TEntity, int> repo, bool needRecords = false)
423 where TEntity : class, IEntity<int>
424 {
425 var ret = new List<object>();
426
427 var defsAsList = ((IEnumerable) definitions).Cast<object>().ToList();
428 foreach (var definition in defsAsList)
429 {
430 var serializeForCreateNewObject = JsonConvert.SerializeObject(definition);
431 var newObject = NewPlayerGeneratorHelper.GetNewObjectByDefinition<TEntity>(serializeForCreateNewObject,
432 plantation.PlantationStorage.Id, entityEnum, plantation.PlayerStorage.Id);
433
434 if (needRecords)
435 {
436 var id = await repo.InsertAndGetIdAsync(newObject);
437 var newItem = await repo.GetAsync(id);
438 ret.Add(newItem);
439 }
440 else
441 await repo.InsertAsync(newObject);
442 }
443
444 return ret;
445 }
