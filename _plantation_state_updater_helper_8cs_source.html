116 public static void SendItemsForNewLvl(AnalysisState state)
117 {
118 var receivedLevels = state.ReceivedLevels;
119 if (receivedLevels <= 0) return;
120
121 var storage = state.StorageEntity;
122 if (storage is PlantationStorage plantationStorage)
123 {
124 var allItemsNames = GenerateNamesForNewLvlItems(state.ReceivedLvlItems);
125 if (!allItemsNames.Any()) return;
126
127 var currLvlQuestsData = state.ReceivedLvlItems.Quests.Select(entity => new LvlQuestsData
128 {
129 QuestId = entity.Id,
130 QuestType = entity.QuestType
131 }).ToList();
132
133 var responseObj = new ItemsForNewLvls
134 {
135 ItemsNames = allItemsNames,
136 QuestsData = currLvlQuestsData,
137 S2DistrictsList = null
138 };
139
140 state.InfoHub.Clients.User(plantationStorage.UserId.ToString()).itemsForNewLvls(responseObj);
141 }
142 else if (storage is PlayerStorage playerStorage)
143 {
144 if (!state.NewDistricts.Any()) return;
145
146 var parser = state.DisplayParserStrategy;
147 var currLvlDistrictsNames = state.NewDistricts.Select(entity =>
148 "Dostępna nowa dzielnica - " + entity.Name +
149 ". Start: " +
150 parser.ParseValue(EnumUtils.PropTypes.DateTime, entity.StartTime) + " Koniec: " +
151 parser.ParseValue(EnumUtils.PropTypes.DateTime, entity.EndTime)
152 ).ToList();
153
154 var responseObj = new ItemsForNewLvls
155 {
156 ItemsNames = currLvlDistrictsNames,
157 QuestsData = null,
158 S2DistrictsList = state.S2DistrictsList
159 };
160
161 state.InfoHub.Clients.User(playerStorage.UserId.ToString()).itemsForNewLvls(responseObj);
162 }
163 }
