78 public static AnalysisState UpdateStorageState(this AnalysisState state)
79 {
80 if (!state.InitStatus || state.LastEntityPropertyChanges == null) return state;
81
82 var entity = state.StorageEntity;
83 var displayParser = state.DisplayParser;
84 var entityType = state.Entity.GetType();
85 var properties = entityType.GetProperties().ToList();
86
87 var levelProp = properties.FirstOrDefault(item => item.Name == PlantationStorageObservedFields.Level);
88 var gainedExperience = properties.FirstOrDefault(item => item.Name == PlantationStorageObservedFields.GainedExperience);
89 var gold = properties.FirstOrDefault(item => item.Name == PlantationStorageObservedFields.Gold);
90
91 var parsedLevel = (string) displayParser.Parse(levelProp, entity.Level);
92 var parsedGainedExperience = (string) displayParser.Parse(gainedExperience, entity.GainedExperience);
93 var parsedGold = (string) displayParser.Parse(gold, entity.Gold);
94
95 if (state.Entity is PlantationStorage plantationStorage)
96 {
97 if (state.PlayerStorage.LastSelectedDistrictId != plantationStorage.DistrictId) return state;
98
99 PlantationStateUpdaterHelper.SendItemsForNewLvl(state);
100 var obj = PlantationStateUpdaterHelper.ProcessPlantationStorageUpdate(displayParser,
101 plantationStorage, parsedLevel, parsedGainedExperience, parsedGold, state.ReceivedLevels);
102 state.InfoHub.Clients.User(state.UserId.ToString()).updatePlantationState(obj);
103 }
104 else if (state.Entity is PlayerStorage playerStorage)
105 {
106 PlantationStateUpdaterHelper.SendItemsForNewLvl(state);
107
108 var honor = properties.FirstOrDefault(item => item.Name == PlayerStorageObservedFields.Honor);
109 var parsedHonor = (string) displayParser.Parse(honor, playerStorage.Honor);
110
111 var obj = new PlayerStorageUpdate
112 {
113 Level = parsedLevel,
114 Honor = parsedHonor,
115 GainedExperience = parsedGainedExperience,
116 Gold = parsedGold
117 };
118
119 state.InfoHub.Clients.User(playerStorage.UserId.ToString()).updatePlayerState(obj);
120 }
121
122 return state;
123 }

130 public static AnalysisState UpdateBlackMarketState(this AnalysisState state)
131 {
132 var entity = state.BlackMarketEntity;
133 var blackMarketHub = state.BlackMarketHub;
134 if (state.Action == UpdateActivityOperator.ActionCreated)
135 {
136 var blackMarketItem = state.DisplayParser
137 .Parse(state.ObjectMapper.Map<List<BlackMarketItem>>(new List<BlackMarketTransaction> {entity}).Cast<object>().ToList()).Single();
138
139 if (entity.SellerId != null)
140 {
141 var userConnections = state.OnlineClientManager.GetAllClients()
142 .Where(item => item.UserId == entity.SellerId).Select(item => item.ConnectionId).ToArray();
143
144 blackMarketHub.Clients.User(entity.SellerId.ToString()).setIssuedTransaction(blackMarketItem);
145 blackMarketHub.Clients.AllExcept(userConnections).setTransaction(blackMarketItem);
146 }
147 else
148 blackMarketHub.Clients.All.setTransaction(blackMarketItem);
149 }
150
151 if (state.Action == UpdateActivityOperator.ActionDeleted)
152 {
153 if (entity.SellerId != null)
154 {
155 var userConnections = state.OnlineClientManager.GetAllClients()
156 .Where(item => item.UserId == entity.SellerId).Select(item => item.ConnectionId).ToArray();
157
158 blackMarketHub.Clients.AllExcept(userConnections).removeTransaction(entity.Id, entity.ItemEntityName);
159
160 var sellerMessageData = new BlackMarketTransactionStatus
161 {
162 DistrictId = state.District.Id,
163 Status = BlackMarketStatuses.SuccessStatus,
164 Message = (entity.IsCanceled ? "Anulowano" : "Sprzedano") + ": " +
165 entity.Quantity.ToString(CultureInfo.InvariantCulture).Replace(".", ",") +
166 PlantationManagerHelper.GetMeasureUnitByEntityName(entity.ItemEntityName) + " " + entity.ItemName + (entity.IsCanceled
167 ? "" : ". Zysk: " + (entity.Quantity * entity.Price).ToString("0.##").Replace(".", ",") + "$")
168 };
169
170 blackMarketHub.Clients.User(entity.SellerId.ToString()).issue(sellerMessageData);
171 blackMarketHub.Clients.User(entity.SellerId.ToString()).removeIssuedTransaction(entity.Id, entity.ItemEntityName);
172 }
173 else
174 blackMarketHub.Clients.All.removeTransaction(entity.Id, entity.ItemEntityName);
175 }
176
177 return state;
178 }
