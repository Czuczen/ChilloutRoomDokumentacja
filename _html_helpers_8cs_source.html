84 public static MvcHtmlString GenerateDropDownField(this HtmlHelper helper, PropertyInfo prop, EditOrCreateViewModel model,
85 string value = null, string fieldAttr = null)
86 {
87 var ret = new MvcHtmlString("");
88
89 if (prop == null) return ret;
90
91 var fieldName = prop.Name;
92 var additionalAttr = new RouteValueDictionary();
93 additionalAttr.Add("class", "validate form-control");
94
95 var required = prop.CustomAttributes?.SingleOrDefault(currAttr =>
96 currAttr?.AttributeType?.FullName?.Contains("FieldIsRequiredAttribute") ?? false) != null;
97
98 if (required)
99 additionalAttr.Add("required", "required");
100
101 var isMany = model?.IsManyEdit != null && model.IsManyEdit;
102 if (isMany)
103 additionalAttr.Add("disabled", "true");
104
105 if (fieldAttr == "type=checkbox")
106 {
107 var list = SelectListLoaderHelper.GetCheckBoxAsSelectOptions(value);
108 ret = helper.DropDownList(fieldName, list, additionalAttr);
109 }
110 else
111 {
112 ret = helper.DropDownList(fieldName, (List<SelectListItem>) model?.FieldsOptions[fieldName], "", additionalAttr);
113 }
114
115 return ret;
116 }

180

182 model.CurrValue = propValue;
183
184 var condition = ViewHelper.CheckInputRenderConditions(propName, model);
185

188 else if (condition != null)
189 ret += $"<input id='{propName}' type='hidden' name='{propName}' value='{propValue}' class='validate form-control' required>";
