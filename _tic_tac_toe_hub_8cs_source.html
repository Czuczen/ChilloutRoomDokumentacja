106 public async Task SetInQueueRoom()
107 {
108 try
109 {
110 var groupWasFound = false;
111
112 var currClient = _onlineClientManager.GetByConnectionIdOrNull(Context.ConnectionId);
113 var allClients = GetConnectedClients();
114 var currClientTicTacToeData = (TicTacToeClientData) currClient.Properties[GameDataKey];
115
116 foreach (var currOpponent in allClients)
117 {
118 try
119 {
120 if (currClient.UserId == currOpponent.UserId) continue;
121
122 var opponentTicTacToeData = (TicTacToeClientData) currOpponent.Properties[GameDataKey];
123 if (opponentTicTacToeData.WaitForOpponent)
124 {
125 opponentTicTacToeData.WaitForOpponent = false;
126
127 currClientTicTacToeData.OpponentConnectionId = currOpponent.ConnectionId;
128 opponentTicTacToeData.OpponentConnectionId = Context.ConnectionId;
129
130 await Clients.Client(currOpponent.ConnectionId).opponentFound(currClient.ConnectionId);
131 await Clients.Client(currClient.ConnectionId).opponentFound(currOpponent.ConnectionId);
132
133 groupWasFound = true;
134 }
135 }
136 catch (Exception ex)
137 {
138 Logger.Error("Blad=====/=====", ex);
139 await Clients.Client(Context.ConnectionId).errorOccured(ex.Message);
140 }
141 }
142
143 if (!groupWasFound)
144 {
145 currClientTicTacToeData.WaitForOpponent = true;
146 Clients.Caller.mustWaitForOpponent();
147 }
148 }

212 public virtual async Task StartMatch(string opponentConnectionId)
213 {
214 try
215 {
216 var currClient = _onlineClientManager.GetByConnectionIdOrNull(Context.ConnectionId);
217 var currClientStorage = await _ticTacToeStorageRepository.SingleAsync(item => item.UserId == currClient.UserId);
218 var currClientTicTacToeData = (TicTacToeClientData) currClient.Properties[GameDataKey];
219
220 var currOpponent = _onlineClientManager.GetByConnectionIdOrNull(opponentConnectionId);
221 var opponentTicTacToeData = (TicTacToeClientData) currOpponent?.Properties[GameDataKey];
222
223 currClientTicTacToeData.WantToPlayWithOpponent = true;
224 if (opponentTicTacToeData != null && opponentTicTacToeData.WantToPlayWithOpponent)
225 {
226 var groupName = Guid.NewGuid().ToString();
227 opponentTicTacToeData.GroupName = groupName;
228 currClientTicTacToeData.GroupName = groupName;
229 var opponentStorage = await _ticTacToeStorageRepository.SingleAsync(item => item.UserId == currOpponent.UserId);
230
231 var symbolNumber = _random.Next(1, 3);
232 var currPlayerSymbol = symbolNumber == 1 ? "X" : "O";
233 var opponentSymbol = symbolNumber == 2 ? "X" : "O";
234
235 currClientTicTacToeData.OpponentUserId = currOpponent.UserId;
236 opponentTicTacToeData.OpponentUserId = currClient.UserId;
237
238 currClientTicTacToeData.GameInProgress = true;
239 opponentTicTacToeData.GameInProgress = true;
240
241 currClientTicTacToeData.Symbol = currPlayerSymbol;
242 opponentTicTacToeData.Symbol = opponentSymbol;
243
244 var currPlayerData = new TicTacToeStartMatchData();
245 var currPlayerMappedData = _objectMapper.Map(currClientStorage, currPlayerData);
246 currPlayerMappedData.Symbol = currPlayerSymbol;
247
248 var opponentData = new TicTacToeStartMatchData();
249 var opponentMappedData = _objectMapper.Map(opponentStorage, opponentData);
250 opponentMappedData.Symbol = opponentSymbol;
251
252 opponentTicTacToeData.WantToPlayWithOpponent = false;
253 currClientTicTacToeData.WantToPlayWithOpponent = false;
254
255 await Groups.Add(currOpponent.ConnectionId, groupName);
256 await Groups.Add(Context.ConnectionId, groupName);
257
258 await Clients.Group(groupName, Context.ConnectionId).matchStarted(groupName, currPlayerMappedData);
259 await Clients.Group(groupName, currOpponent.ConnectionId).matchStarted(groupName, opponentMappedData);
260 }
261 else if (opponentTicTacToeData != null)
262 {
263 await Clients.Client(Context.ConnectionId).mustWaitForOpponentChoice();
264 }
265 else
266 {
267 currClientTicTacToeData.WantToPlayWithOpponent = false;
268 currClientTicTacToeData.WaitForOpponent = true;
269 await Clients.Client(Context.ConnectionId).opponentDisconnectedFromQueue();
270 }
271 }
272 catch (Exception ex)
273 {
274 Logger.Error("Blad=====/=====", ex);
275 await Clients.Client(Context.ConnectionId).errorOccured(ex.Message);
276 }

312 {
313 var ret = new TicTacToeEndMatchData();
314 try
315 {
316 TicTacToeStorage storage;
317 if (!fromBackend)
318 {
319 var currClient = _onlineClientManager.GetByConnectionIdOrNull(Context.ConnectionId);
320 var ticTacToeData = (TicTacToeClientData) currClient.Properties[GameDataKey];
321 await Groups.Remove(Context.ConnectionId, ticTacToeData.GroupName);
322
323 ticTacToeData.GameInProgress = false;
324 ticTacToeData.OpponentConnectionId = null;
325 ticTacToeData.GroupName = null;
326 ticTacToeData.OpponentUserId = null;
327 ticTacToeData.Symbol = null;
328 ticTacToeData.WantToPlayWithOpponent = false;
329 ticTacToeData.WaitForOpponent = false;
330
331 storage = await _ticTacToeStorageRepository.SingleAsync(item => item.UserId == currClient.UserId);
332 }
333 else
334 {
335 storage = await _ticTacToeStorageRepository.SingleAsync(item => item.UserId == opponentUserId);
336 }
337
338 storage.GamesPlayed++;
339 switch (matchResult)
340 {
341 case Tied:
342 storage.TiedGames++;
343 break;
344 case Win:
345 storage.GamesWon++;
346 break;
347 case Lose:
348 storage.GamesLost++;
349 break;
350 }
351
352 ret = _objectMapper.Map(storage, ret);
353 }
354 catch (Exception ex)
355 {
356 Logger.Error("Blad=====/=====", ex);
357 await Clients.Client(Context.ConnectionId).errorOccured(ex.Message);
358 }
