165 // Kiedy opiekun dzielnicy edytuje zadanie które jest już nie dostępne czasowo ale jest ukończone to z racji że zostało zresetowane musi zostać usunięte z listy/akordeon ukończonych. Jeśli ustawimy WorkerSent na true to RemoveNotAvailableEventQuests ustawi zadanie tam gdzie jego miejsce.

175 var questRequirementsProgress = JsonConvert.DeserializeObject<Dictionary<int, decimal>>(questProgress.RequirementsProgress);
176 var newQuestRequirementsProgress = new Dictionary<int, decimal>();
177
178 foreach (var reqProgress in questRequirementsProgress)
179 {
180 newQuestRequirementsProgress[reqProgress.Key] = reqProgress.Value;
181
182 var req = requirements.Single(item => item.Id == reqProgress.Key);
183 if (req.Condition != DbRequirementConditionsNames.Possession &&
184 req.Condition != DbRequirementConditionsNames.Deliver) continue;
185
186 if (!string.IsNullOrWhiteSpace(req.CustomEntityName))
187 {
188 switch (req.CustomEntityName)
189 {
190 case DbCustomEntityNames.PlantationStorage:
191 var tempValue1 = plantationStorage.GetType().GetProperty(req.Comparer).GetValue(plantationStorage);
192 var value1 = tempValue1 == null || tempValue1.ToString() == "" ? 0 : decimal.Parse(tempValue1.ToString().Replace(".", ","));
193 newQuestRequirementsProgress[req.Id] = value1 > req.Amount ? req.Amount : value1;
194 break;
195 case DbCustomEntityNames.PlayerStorage:
196 var tempValue2 = playerStorage.GetType().GetProperty(req.Comparer).GetValue(playerStorage);
197 var value2 = tempValue2 == null || tempValue2.ToString() == "" ? 0 : decimal.Parse(tempValue2.ToString().Replace(".", ","));
198 newQuestRequirementsProgress[req.Id] = value2 > req.Amount ? req.Amount : value2;
199 break;
200 case DbCustomEntityNames.Seed when req.GeneratedTypeId != null:
201 var seed = seeds.Single(item => item.GeneratedTypeId == req.GeneratedTypeId && item.PlantationStorageId == plantationStorage.Id);
202 var tempValue3 = seed.GetType().GetProperty(req.Comparer).GetValue(seed);
203 var value3 = tempValue3 == null || tempValue3.ToString() == "" ? 0 : decimal.Parse(tempValue3.ToString().Replace(".", ","));
204 newQuestRequirementsProgress[req.Id] = value3 > req.Amount ? req.Amount : value3;
205 break;
206 case DbCustomEntityNames.DriedFruit when req.GeneratedTypeId != null:
207 var driedFruit = driedFruits.Single(item => item.GeneratedTypeId == req.GeneratedTypeId && item.PlantationStorageId == plantationStorage.Id);
208 var tempValue4 = driedFruit.GetType().GetProperty(req.Comparer).GetValue(driedFruit);
209 var value4 = tempValue4 == null || tempValue4.ToString() == "" ? 0 : decimal.Parse(tempValue4.ToString().Replace(".", ","));
210 newQuestRequirementsProgress[req.Id] = value4 > req.Amount ? req.Amount : value4;
211 break;
212 }
213 }
214 else if (req.GeneratedTypeId != null)
215 {
216 var generatedType = generatedTypes.Single(item => item.Id == (int) req.GeneratedTypeId);
217 var repo = _customRepositoryLoader.GetRepository(generatedType.EntityName);
218 var entity = repo.GetWhere(RelationFieldsNames.PlantationStorageId, playerDefinition.PlantationStorageId)?
219 .Cast<IGeneratedEntity>()
220 .SingleOrDefault(item => item.GeneratedTypeId == generatedType.Id);
221
222 if (entity == null) continue;
223 var type = entity.GetType();
224 var tempValue = type.GetProperty(req.Comparer).GetValue(entity);
225 var value = tempValue == null || tempValue.ToString() == "" ? 0 : decimal.Parse(tempValue.ToString().Replace(".", ","));
226 newQuestRequirementsProgress[req.Id] = value > req.Amount ? req.Amount : value;
227 }
228 }
229
230 questProgress.RequirementsProgress = JsonConvert.SerializeObject(newQuestRequirementsProgress);
231 }

359 private async Task<List<ExtendingModels.Models.General.Quest>> CreateQuestConnections(int entityId, Dictionary<string, List<int>> entitiesIds)
360 {
361 var questRequirementsProgress = new Dictionary<int, decimal>();
362 var quest = await _questRepository.GetAsync(entityId);
363 var generatedType = await _generatedTypeRepository.GetAsync(quest.GeneratedTypeId);
364
365 var generatedTypeQuery = _generatedTypeRepository.GetAll().Where(item => item.DistrictId == generatedType.DistrictId);
366 var questsProgress = await _questRequirementsProgressRepository.GetAll().Join(
367 _questRepository.GetAll().Join(generatedTypeQuery,
368 currQuest => currQuest.GeneratedTypeId,
369 currGeneratedType => currGeneratedType.Id,
370 (currQuest, currGeneratedType) => currQuest),
371
372 progress => progress.QuestId,
373 currQuest => currQuest.Id,
374 (progress, currQuest) => progress).ToListAsync();
375
376 foreach (var entityIdsPair in entitiesIds)
377 {
378 switch (entityIdsPair.Key)
379 {
380 case EntitiesDbNames.Drop:
381 foreach (var dropId in entityIdsPair.Value)
382 {
383 await _dropQuestRepository.InsertAsync(new DropQuest
384 {DropId = dropId, QuestId = quest.Id});
385 }
386 break;
387 case EntitiesDbNames.Requirement:
388 foreach (var requirementId in entityIdsPair.Value)
389 {
390 questRequirementsProgress[requirementId] = 0;
391 }
392 break;
393 }
394 }
395
396 var newRequirementsProgress = JsonConvert.SerializeObject(questRequirementsProgress);
397 var questProgress = await _questRequirementsProgressRepository.FirstOrDefaultAsync(item => item.QuestId == quest.Id);
398 if (questProgress == null)
399 {
400 await _questRequirementsProgressRepository.InsertAndGetIdAsync(new QuestRequirementsProgress
401 {QuestId = quest.Id, RequirementsProgress = newRequirementsProgress});
402 }
403 else
404 questProgress.RequirementsProgress = newRequirementsProgress;
405
406 var playersDefinitions = await _questRepository.GetAllListAsync(item => item.GeneratedTypeId == quest.GeneratedTypeId && item.Id != entityId);
407 foreach (var playerDefinition in playersDefinitions)
408 {
409 var playerDefinitionProgress = questsProgress.SingleOrDefault(item => item.QuestId == playerDefinition.Id);
410 if (playerDefinitionProgress == null)
411 {
412 await _questRequirementsProgressRepository.InsertAndGetIdAsync(new QuestRequirementsProgress
413 {QuestId = playerDefinition.Id, RequirementsProgress = newRequirementsProgress});
414 }
415 else
416 playerDefinitionProgress.RequirementsProgress = newRequirementsProgress;
417 }
418
419 return playersDefinitions;
420 }
